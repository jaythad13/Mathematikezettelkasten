\begin{problem}[S \& S \S 8.4 \#7(b)]
	Recall that for $\lvert z \rvert < 1$,
	\[
		\log \left ( \frac{1}{1 - z} \right ) = \sum_{k \ge 1} \frac{z^{k}}{k}.
	\]
	We have seen that
	\[
		e^{\log (1 / (1 - z))} = \frac{1}{1 - z}.
	\]
	Show that if $\operatorname{Re} w > 1 /2$ and $w = \rho e^{i \varphi}$ with $\rho > 0$, $\lvert \varphi \rvert < \pi$, then
	\[
		\log w = \log \rho + i \varphi.
	\]
\end{problem}
