For this week's set, let $X = \operatorname{Spec} \mathcal{O}$ for an integral domain $\mathcal{O}$. The \emph{quotient field} of $\mathcal{O}$ is the smallest field containing it, namely $K = \{ p/q \mid p, q \in \mathcal{O}, q \neq 0 \}$. For any Zariski open $U \subset X$, define the integral domain $\mathcal{O}_{X}(U) = \bigcap_{I \in U} \mathcal{O}_{I}$ in terms of the local rings $\mathcal{O}_{I} = \{ p / q \in K \mid q \notin I \}$.

\begin{problem}[Problem 1.]
	We say that subset $V \subset X$ is \emph{Zariski closed} if $V = Z_{J} = \{ I \in \operatorname{Spec} \mathcal{O} \mid J \subset I \} \cong \operatorname{Spec} \mathcal{O} / J$ for some ideal $J \subset \mathcal{O}$. We say that a subset $U \subset X$ is \emph{Zariski open} if $V = X \setminus U$ is Zariski closed.
	\begin{enumerate}
		\item Show that both $\O$ and $X$ are Zariski open.
		\item Say that $\{ U_{\alpha} \mid \alpha \in \mathcal{I} \}$ is an arbitrary family of Zariski open sets. Show that $U = \bigcup_{\alpha \in \mathcal{I}} U_{\alpha}$ is also Zariski open.
		\item Say that $\{ U_{\alpha} \mid \alpha \in \mathcal{I} \}$ is a \emph{finite} family of Zariski open sets. Show $U = \bigcap_{\alpha \in \mathcal{I}} U_{\alpha}$ is also Zariski open.
	\end{enumerate}
\end{problem}

\begin{solution}
	\begin{enumerate}
		\item Note that $\O$ is the set of primes $I \in \operatorname{Spec} \mathcal{O}$ that contain any maximal $M \in \operatorname{Spec} \mathcal{O}$ since, by definition, there are none. Thus, $\O$ is Zariski closed and its complement, $X$, is Zariski open.
			Similarly, note that $X = \operatorname{Spec} \mathcal{O} / \left < 0 \right >$, and thus, $X$ is Zariski closed too. Thus, its complement, $\O$, is Zariski open.
		\item To show that the arbitrary union of open sets is open, it suffices to show its complement is closed. By De Morgan's laws, an arbitrary union of open sets $U_{\alpha}$ has complement
			\[
			X \setminus \bigcup_{\alpha \in \mathcal{I}} U_{\alpha} = \bigcap_{\alpha \in \mathcal{I}} X \setminus U_{\alpha}
			\]
			which is an intersection of closed sets.

			Write $V_{\alpha} = X \setminus U_{\alpha}$ for each $\alpha \in I$ and $V$ for their intersection. By definition, $V_{\alpha} = \operatorname{Spec} \mathcal{O} / J_{\alpha}$ for some ideal $J_{\alpha} \subset \mathcal{O}$. Consider $J = \left < J_{\alpha} \right >_{\alpha \in \mathcal{I}}$ the ideal consisting of all finite linear combinations of $x_{\alpha} \in J_{\alpha}$. We claim $V = \operatorname{Spec} \mathcal{O} / J$.

			Suppose $I \in V$. Then $J_{\alpha} \subset I$ for all $J_{\alpha}$. Since $I$ is additively, closed this gives us all sums $\sum x_{\alpha} \in I$, and thus, $J \subset I$. Conversely, suppose $I \in \operatorname{Spec} \mathcal{O} / J$. Since $J_{\alpha} \subset J \subset I$ for all $J_{\alpha}$, we have $I \in \operatorname{Spec} \mathcal{O} / J_{\alpha}$ for all $J_{\alpha}$. Now we have double containment, and thus, $V = \operatorname{Spec} \mathcal{O} / J$.
		\item To show that the finite intersection of open sets is open, it suffices to show that its complement is closed. By De Morgan's laws, a finite intersection of open sets $U_{\alpha}$ has complement
			\[
				X \setminus \bigcap_{\alpha \in \mathcal{I}} U_{\alpha} = \bigcup_{\alpha \in \mathcal{I}} X \setminus U_{\alpha}
			\]
			which is a finite union of closed sets.

			Write $V_{\alpha} = X \setminus U_{\alpha}$ for each $\alpha \in I$ and $V$ for their intersection. By definition, $V_{\alpha} = \operatorname{Spec} \mathcal{O} / J_{\alpha}$ for some ideal $J_{\alpha} \subset \mathcal{O}$. Consider the product $J = \prod_{\alpha \in \mathcal{I}} J_{\alpha}$. We claim $V = \operatorname{Spec} \mathcal{O} / J$.

			Suppose $I \in V$. Then $J_{\beta} \subset I$ for some $\beta \in \mathcal{I}$. But
			\[
				J = \prod_{\alpha \in \mathcal{I}} J_{\alpha} \subset J_{\beta} \subset I
			\]
			so $I \in \operatorname{Spec} \mathcal{O} / J$.

			Conversely, suppose $I \in \operatorname{Spec} \mathcal{O} / J$. Note that this means $I$ is prime. If any $J_{\alpha} \subset I$, then $I \in \operatorname{Spec} \mathcal{O} / J_{\alpha} \subset V$ and we are done. If we don't know of some $J_{\alpha} \subset I$, we can choose $J_{\alpha} \not \subset I$ for each $\alpha \neq \beta$ (for all $\alpha \in \mathcal{I}$ except one). Then, for each of these $\alpha \neq \beta$, there must be at least one $j_{\alpha} \in J_{\alpha}$ such that $j_{\alpha} \notin I$. If $\alpha_{1}, \dots, \alpha_{n}$ are all the $\alpha \neq \beta$, we can show $J_{\beta} \subset I$ by analysing $j_{\alpha_{1}} \dotsm j_{\alpha_{n}} j_{\beta} \in J \subset I$ for each $j_{\beta} \in J_{\beta}$. Notice that $j_{\alpha_{1}} \dotsm j_{\alpha_{n}} \notin I$ since $I$ is prime and each $j_{\alpha_{k}} \notin I$. But then, since  $I$ is prime with $(j_{\alpha_{1}} \dotsm j_{\alpha_{n}}) j_{\beta} \in I$, we have $j_{\beta} \in I$.
	\end{enumerate}
\end{solution}

\begin{problem}[Problem 2.]
	Denote $\mathcal{O} = \mathbb{Z}$, so that its quotient field is $K = \mathbb{Q}$.
	\begin{enumerate}
		\item Assume that $I = \{ 0 \}$ is the zero ideal. Show that we have the localisation $\mathcal{O}_{I} = \mathbb{Q}$.
		\item Assume that $I = \left < \ell \right > = \ell \mathbb{Z}$ for a rational prime $\ell$. Show that $\mathcal{O}_{I} = \{ p / q \in \mathbb{Q} \mid \ell \text{ does not divide } q \}$.
	\end{enumerate}
\end{problem}

\begin{solution}
	\begin{enumerate}
		\item Since $\mathcal{O} = \mathbb{Z}$ is an integral domain, $I = \{ 0 \}$ is a prime ideal, and $\mathcal{O}_{I}$ is a well defined localisation away from $I$. In particular,
			\[
				\mathcal{O}_{I} = \{ p / q \mid q \notin I \} = \{ p / q \mid q \neq 0 \} = \mathbb{Q}.
			\]
		\item For primes $\ell \in \mathbb{Z}$, $I = \left < \ell \right >$ is a prime ideal of $\mathcal{O} = \mathbb{Z}$ consisting of the multiples of $\ell$. Thus, $\mathcal{O}_{I}$ is a well defined localisation away from $I$. In particular,
			\[
				\mathcal{O}_{I} = \{ p / q \mid q \notin I \} = \{ p / q \mid \ell \text{ does not divide } q \}.
			\]
	\end{enumerate}
\end{solution}

\begin{problem}[Problem 3.]
	Since $\mathcal{O} = \mathbb{Z}$ is a Dedekind domain, we see that $X = \operatorname{Spec} \mathbb{Z}$ is a non-singular affine curve. Say that $V = \{ \ell_{1}, \ell_{2}, \dots, \ell_{n} \} \subset \mathbb{Z}$ is a finite collection of rational primes.
	\begin{enumerate}
		\item Upon identifying $\ell_{\alpha} \in V$ with $\left < \ell_{\alpha} \right > \in X$, show that the complement $U = X \setminus V$ is a Zariski open subset.
		\item Show that $\mathcal{O}_{X}(U) = \bigcap_{I \in U} \mathcal{O}_{I} = \{ p / q \in \mathbb{Q} \mid \text{no } \ell \in U \text{ divides } q \}$.	
	\end{enumerate}
\end{problem}

\begin{solution}
	\begin{enumerate}
		\item Since each $\left < \ell_{\alpha} \right >$ is prime, each $V_{\alpha} = \{ \left < \ell_{\alpha} \right > \}$ is Zariski closed, and each $U_{\alpha} = X \setminus V_{\alpha}$ is Zariski open. By De Morgan's laws
			\[
				U = X \setminus \left ( \bigcup_{\alpha = 1}^{n} V_{\alpha} \right ) = \bigcap_{\alpha = 1}^{n} X \setminus V_{\alpha} = \bigcap_{\alpha = 1}^{n} U_{\alpha}.
			\]
			By the first exercise, this finite intersection of open sets is open.
		\item Since $\mathbb{Z}$ is also a principle ideal domain, every prime ideal is just $\left < \ell \right >$ for some rational prime $\ell \in \mathbb{Z}$. Thus, the prime ideals $I \in U$ are exactly the ideals $\left < \ell \right >$ where $\ell \in \mathbb{Z}$ is a prime not equal to any $\ell_{\alpha} \in V$. 

			By the previous exercise, the localisation away from $I$ is
			\[
				\mathcal{O}_{I} = \{ p / q \in \mathbb{Q} \mid \ell \text{ does not divide } q \}.
			\]
			The intersection of all of these $\mathcal{O}_{I}$ is then exactly the set of fractions $p / q$ where no  $\ell \in I$ for any $I \in U$ divides $q$. But then since the primes in $I$ in $U$ are just the primes in $U$, we have
			\[
				\mathcal{O}_{X}(U) = \bigcap_{I \in U} = \{ p / q \in \mathbb{Q} \mid \text{no } \ell \in U \text{ divides } q \}
			\]
	\end{enumerate}
\end{solution}

\begin{problem}[Problem 4.]
	Denote $\mathcal{O} = F[x]$ as the collection of polynomials $p(x) = a_{n} x^{n} + \dots + a_{1} x + a_{0}$ with coefficients in a field $F$, so that its quotient field $K = F(x)$ consists of rational functions $p(x) / q(x)$.
	\begin{enumerate}
		\item Assume that $I = \{ 0 \}$ is the zero ideal. Show that the localisation $\mathcal{O}_{I} = F(x)$.
		\item Assume that $I = \left < f \right >$ for the linear polynomial $f(x) = x - r$. Show that $q(x) \in I$ if and only if $q(r) = 0$, and conclude that the localisation is $\mathcal{O}_{I} = \{ p(x) / q(x) \in F(x) \mid q(r) \neq 0 \}$.
	\end{enumerate}
\end{problem}

\begin{solution}
	\begin{enumerate}
		\item Since $I = \{ 0 \}$ is a prime ideal, we have the well-defined localisation
			\[
				\mathcal{O}_{I} = \{ p(x) / q(x) \mid q \notin I \} = \{ p(x) / q(x) \mid q(x) \neq 0 \} = F(x).
			\]
		\item Suppose $q(x) \in I$. Then $f$ divides $q$, so $q(r) = f(r) g(r) = 0$. Suppose conversely that $q(r) = 0$. Then by the division algorithm, $f(x) = x - r$  divides $q(x)$ and thus, $q \in I$.

			Since $I = \left < x - r \right >$ is a prime (in fact, it is maximal), we have the well defined localisation
			\[
				\mathcal{O}_{I} = \{ p(x) / q(x) \mid q \notin I \} = \{ p(x) / q(x) \in F(x) \mid q(r) \neq 0 \}
			\]
			where the last equality follows from our earlier result that $I$ contains exactly those polynomials vanishing at $r$.
	\end{enumerate}
\end{solution}

\begin{problem}[Problem 5.]
	Since $\mathcal{O} = F[x]$ is a Dedekind domain, we see that $X = \operatorname{Spec} \mathcal{O} = \mathbb{A}^{1}_{F}$ is a non-singular affine curve. Say that $V = \{ r_{1}, r_{2}, \dots, r_{n} \} \subset F$ is a finite collection of elements.
	\begin{enumerate}
		\item Upon identifying $r_{\alpha} \in V$ with $\left < x - r_{\alpha} \right > \in X$, show the complement $U = X \setminus V$ is a Zariski open subset. Show that $V \cong Z_{J} = \{ I \in X \mid J \subset I \}$ in terms of the ideal $J = \left < (x - r_{1}) (x - r_{2}) \dotsm (x - r_{n}) \right >$.
		\item Show that $V \simeq Z_{J} = \operatorname{Spec} \mathcal{O} / J$ in terms of the ideal $J = \left < (x - r_{1}) (x - r_{2}) \dots (x - r_{n}) \right >$
		\item Show that $\mathcal{O}_{X}(U) = \bigcap_{I \in U} \mathcal{O}_{I} = \{ p(x) / q(x) \in F(x) \mid q(r) \neq 0 \text{ for each } r \in U \}$.
	\end{enumerate}
\end{problem}

\begin{solution}
	\begin{enumerate}
		\item Since each $\left < x - r_{\alpha} \right >$ is prime, each $V_{\alpha} = \{ \left < x - r_{\alpha} \right > \}$ is Zariski closed, and each $U_{\alpha} = X \setminus V_{\alpha}$ is Zariski open. By De Morgan's laws
			\[
				U = X \setminus \left ( \bigcup_{\alpha = 1}^{n} V_{\alpha} \right ) = \bigcap_{\alpha = 1}^{n} X \setminus V_{\alpha} = \bigcap_{\alpha = 1}^{n} U_{\alpha}.
			\]
			By the first exercise, this finite intersection of open sets is open.
		\item In the proof of part (c) of the first problem, we show that the finite union of closed sets $V_{\alpha} = \operatorname{Spec} \mathcal{O} / J_{\alpha}$ is a closed set $V = \operatorname{Spec} \mathcal{O} / J$ where $J = \prod_{\alpha \in \mathcal{I}} J_{\alpha}$. But here $V$ is the finite union of closed sets $V_{\alpha} = \{ \left < x - r_{\alpha} \right > \}$, but since $\left < x - r_{\alpha} \right >$ is maximal, the only prime ideal that contains it is itself --- that is, $V_{\alpha} = \operatorname{Spec} \mathcal{O} / \left < x - r_{\alpha} \right >$. Thus, $J_{\alpha} = \left < x - r_{\alpha} \right >$ and their product $J$ is $\left < (x - r_{1}) \dotsm (x - r_{n}) \right >$ with $V = \operatorname{Spec} \mathcal{O} / J = Z_{J}$ as desired.
		\item The prime ideals $I \in U$ are exactly the ideals $\left < x - r \right >$ where $r \neq r_{\alpha}$ for any $r_{\alpha} \in V$	. 

			By the previous exercise, the localisation away from $I = \left < x - r \right >$ is
			\[
				\mathcal{O}_{I} = \{ p / q \in \mathbb{Q} \mid q(r) \neq 0 \}.
			\]
			The intersection of all of these $\mathcal{O}_{I}$ is then exactly the set of rational functions $p(x) / q(x)$ where $q(r) \neq 0$ for any $r$ with $\left < x - r \right > \in U$. That is,
			\[
				\mathcal{O}_{X}(U) = \bigcap_{I \in U} \mathcal{O}_{I} = \{ p(x) / q(x) \in F(x) \mid q(r) \neq 0 \text{ for each } \left < x - r \right > \in U \}
			\]
			which is exactly the desired result under our identification of $r$ with $\left < x - r \right >$.
	\end{enumerate}
\end{solution}
