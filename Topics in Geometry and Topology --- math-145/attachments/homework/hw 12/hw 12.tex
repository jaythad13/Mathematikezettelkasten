\begin{problem}[67. (The 5 lemma)]
	In the following commutative diagram of vector spaces and linear maps
	\[
		\begin{tikzcd}
			A \ar[r, "p"] \ar[d, "a"] & B \ar[r, "q"] \ar[d, "b"] & C \ar[r, "r"] \ar[d, "c"] & D \ar[r, "s"] \ar[d, "d"] & E \ar[d, "e"] \\
			A' \ar[r, "p'"] & B \ar[r, "q'"] & C \ar[r, "r'"] & D \ar[r, "s'"] & E
		\end{tikzcd}
	\]
	the two rows are exact and the maps $a, b, d, e$ are isomorphisms.
	\begin{enumerate}
		\item Show that $c$ is an isomorphism.

		\item In fact, the maps $a, e$ need not be isomorphisms. Identify weaker conditions on $a, e$ (with $b, d$ still isomorphisms) that are sufficient to deduce that $c$ is an isomorphism.
	\end{enumerate}
\end{problem}

\begin{problem}[68.]
	Consider the simplicial complexes
	\begin{align*}
		B & = \left < 125, 256, 236, 367, 347, 478, 148, 158 \right > & \text{the band} \\
		M & = \left < 125, 256, 236, 367, 347, 478, 458, 158 \right > & \text{the M\"obius band}
	\end{align*}
	and their subcomplexes
	\begin{align*}
		B_{\text{top}} & = \left < 12, 23, 34, 41 \right > & \text{top edge of the band} \\
		B_{\text{both}} & = \left < 12, 23, 34, 41, 56, 67, 78, 85 \right > & \text{both edges of the band} \\
		M_{\text{edge}} & = \left < 12, 23, 34, 45, 56, 67, 78, 81 \right > & \text{the edge of the M\"obius band}
	\end{align*}

	\begin{enumerate}
		\item Sketch the complexes $B$ and $M$, and also the pairs of complexes $(B, B_{\text{top}})$, $(B, B_{\text{both}})$, and $(M, M_{\text{edge}})$. Sketch them as continuous topological objects rather than as simplicial complexes with corners and edges.

		\item Consider the Betti vectors and relative Betti vectors
			\[
				b(B), \quad b(M), \quad b(B, B_{\text{top}}), \quad b(B, B_{\text{both}}), \quad b(M, M_{\text{edge}}).
			\]
			
			Calculate all five Betti vectors three times: using coefficients in $\mathbb{F}_{2}$, $\mathbb{F}_{3}$, and $\mathbb{F}_{47}$.
	\end{enumerate}
\end{problem}
