\begin{problem}[7.1, (3)  $\implies$ (4)]
	Let $X$ and $Y$ be topological spaces and let $f : X \to Y$ be a function. If for every limit point $p$ of a set $A$ in $X$, the image $f(p)$ belongs to $\overline{f^{\text{img}}(A)}$, then for every $x \in X$ and open set $V$ containing $f(x)$, there is an open set $U$ containing $x$ such that $f^{\text{img}}(U) \subseteq V$.
\end{problem}

\pagebreak

\begin{problem}[7.7]
	Let $X$ be a space with a dense set $D$, and let $Y$ be Hausdorff. Let $f : X \to Y$ and $g : X \to Y$ be continuous functions such that for every $d$ in $D$, $f(d) = g(d)$. Then for all $x \in X$, $f(x) = g(x)$
\end{problem}

\begin{problem}[7.24]
	Let $X$ be compact and let $Y$ be Hausdorff. Then any continuous function $f : X \to Y$ is closed.
\end{problem}

\begin{problem}[7.41]
	Let $\mathbb{R}^{\omega}$ be the countably infinite product of $\mathbb{R}$ with itself. Let $f : \mathbb{R} \to \mathbb{R}^{\omega}$ be defined by $f(x) = (x, x, x, \dots)$. Then $f$ is continuous if $\mathbb{R}^{\omega}$ is given the product topology, but not if given the box topology.
\end{problem}
