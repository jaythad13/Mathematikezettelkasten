\begin{problem}[(1)]
	Prove \emph{Sylvester's theorem}: given a quadratic form $V, q$ over $\mathbb{R}$, with a decomposition $V, q \cong \left < 1 \right >^{\oplus r} \oplus \left < -1 \right >^{\oplus s}$, the integers $r, s$ are invariants of $V, q$. In particular, $r$ is the dimension of any maximal subspace of $V$ on which $q$ is positive definite.
\end{problem}

\begin{solution}
	Suppose $\left < 1 \right >^{\oplus r_{1}} \oplus \left < -1 \right >^{\oplus s_{1}} \cong \left < 1 \right >^{\oplus r_{2}} \oplus \left < -1 \right >^{\oplus s_{2}}$ (given by an isomorphism $T : V_{1}, q_{1} \to V_{2}, q_{2}$). We claim that $r_{1} = r_{2}$ and $s_{1} = s_{2}$.

	Since the vector spaces must have equal dimension, $r_{1} + s_{1} = r_{2} + s_{2}$. Thus, it suffices to show $r_{1} = r_{2}$. We claim that the image of $U_{1} = \left < 1 \right >^{\oplus r_{1}}$ is contained in $\left < 1 \right >^{\oplus r_{2}}$. Else, $T^{\text{img}}(U_{1})$ would contain some subspace of $\left < -1 \right >^{\oplus s_{2}}$ on which $q_{2}$ is negative. This would force $q_{1}$ to be negative on $U_{1}$, which is false. 

	Suppose, without loss of generality, that $r_{1} \ge r_{2}$. Since $T$ is an isomorphism, $T_{\mid \left < 1 \right >^{\oplus r_{1}}} : \left < 1 \right >^{\oplus r_{1}} \to \left < 1 \right >^{\oplus r_{2}}$ is an injection from a higher-dimensional vector space to a lower-dimensional one, and thus, also an isomorphism. Thus, $r_{1} = r_{2}$.
\end{solution}

\begin{problem}[(2)]
	Let $V, q$ be a quadratic space of dimension $n$. Show that any element of the orthogonal group is a product of reflections.
\end{problem}

\begin{solution}
	We claim that the group generated by reflections acts transitively on ``spheres'' --- sets of the form $\{ v \in V \mid q(v) = a \}$ for $a \in \mathbb{F}^{\times}$. Suppose $v, w \in V$ have the same nonzero norm $q(v) = q(w) = a$ and let $u_{\pm} = v \pm w$. Note that $\beta(u_{+}, u_{-}) = 0$. Thus, $2 q(v) = q(u_{+}) + q(u_{-})$, and one of $u_{\pm}$ must be anisotropic. Using the fact that $2v = u_{+} + u_{-}$, we can show that $R_{- u_{+}} v = w$ if $u_{+}$ is anisotropic and $R_{u_{-}} v = w$ if $u_{-}$ is anisotropic.

	Now we induct on the dimension of $V$ to get the desired result. If $\dim V = 1$, then a quadratic linear isomorphism $T \in \mathrm{O}(V, q)$ is determined by a choice of $v \mapsto w$ for any non-zero $v, w \in V$ with $q(v) = q(w)$. But there is a reflection $R_{u}$ with $v \mapsto w$. Thus, $T = R_{u}$.

	Suppose $\dim V = n$. Diagonalise $V, q$ as $\left < a_{1}, \dots, a_{n - 1} \right > \oplus \left < a_{n} \right > = U \oplus \left < a_{n} \right >$ and let $e_{1}, \dots, e_{n}$ be the corresponding basis. For $T \in \mathrm{O}(V, q)$, let $R$ be a reflection with $R T e_{n} = e_{n}$. Thus $RT_{\mid U}$ is an isomorphism. Since $RT$ maps $\left < a_{n} \right >$ to $\left < a_{n} \right >$ isomorphically, it must also map orthogonal complements $U$ into $U$ isomorphically. By induction, $(RT)_{\mid U} : U \to U$ is a product of reflections $S_{u_{i}} \in \mathrm{O}(U, q_{\mid U})$ through $u_{i} \in U$. The reflections $R_{u_{i}} \in \mathrm{O}(V, q)$ restrict to $R_{u_{i} \mid U} = S_{u_{i}}$ and have $R_{u_{i}} e_{n} = e_{n}$ (since $e_{n}$ is orthogonal to $U$). Thus, $RT$ is a product of all the $R_{u_{i}}$. Since $R^{-1} = R$, we obtain that $T$ is a product of reflections.
\end{solution}

\begin{problem}[(3)]
	Let $V, q$ be a quadratic space. Show that the direct sum $(V, q) \oplus (V, -q)$ is hyperbolic --- isomorphic to the direct sum of copies of the hyperbolic plane $\left < 1, -1 \right >$.
\end{problem}

\begin{solution}
	Diagonalise to get $(V, q) \cong \left < a_{1}, \dots, a_{n} \right >$. Then $(V, q) \oplus (V, -q) \cong \left < a_{1}, -a_{1} \right > \oplus \dots \oplus \left < a_{n}, -a_{n} \right >$ by a permutation of basis.

	For all squares $a$, we have an isomorphism $\left < a, -a \right > \to \left < 1, -1 \right >$ given by scaling by $\sqrt{a}$. It remains to give an isomorphism $\left < a, -a \right > \cong \left < 1, -1 \right >$ when $a$ is not a square. Consider
	\[
		\left < a(1 - u^{2}), \frac{- a u^{2}}{1 - u^{2}} \right >
	\]
\end{solution}

\begin{problem}[(4)]
	Let $\alpha, \beta \in \mathbb{F}^{\times}$ with $\alpha^{2} + \beta^{2} \neq 0$. Show that the quadratic form $V, q \cong \left < 1, -(\alpha^{2} + \beta^{2}) \right >$ has the property that the direct sum $(V, q) \oplus (V, q)$ is hyperbolic. Further, show that $V$ is hyperbolic if and only if $\alpha^{2} + \beta^{2}$ is a square in $\mathbb{F}$.
\end{problem}

\begin{solution}
	Suppose $\alpha^{2} + \beta^{2}$ is a square. Then $V, q$ is already hyperbolic (just scale $e_{2}$ by $\sqrt{\alpha^{2} + \beta^{2}}$ to obtain the isomorphism $V, q \cong \left < 1, -1 \right >$). It follows that $(V, q) \oplus (V, q)$ is hyperbolic.

	Suppose $\alpha^{2} + \beta^{2}$ is not a square. Then $(V, q) \oplus (V, q) \cong \left < 1, 1, -(\alpha^{2} + \beta^{2}), -(\alpha^{2} + \beta^{2}) \right >$.

	Suppose $V, q$ is hyperbolic. Then there exist linearly independent $v, w \in \mathbb{F}^{2}$ such that $v_{1} w_{1} - v_{2} w_{2} = 0$, $v_{1}^{2} - v_{2}^{2} = 1$, and $w_{1}^{2} - w_{2}^{2} = - (\alpha^{2} + \beta^{2})$. Note that
	\begin{align*}
		-(\alpha^{2} + \beta^{2}) & = (v_{1}^{2} - v_{2}^{2}) (w_{1}^{2} - w_{2}^{2}) \\
					  & = v_{1}^{2} w_{1}^{2} - v_{1}^{2} w_{2}^{2} - v_{2}^{2} w_{1}^{2} + v_{2}^{2} w_{2}^{2} \\
					  & = (v_{1} w_{1} + v_{1} w_{2})(v_{1} w_{1} - v_{1} w_{2}) + (v_{2} w_{2} + v_{2} w_{1})(v_{2} w_{2} - v_{2} w_{1})
	\end{align*}
\end{solution}

\begin{problem}[(5)]
	Suppose $d \in \mathbb{F}^{\times}$. Show that the collection of non-zero elements represented by the quadratic form $\left < 1, d \right >$ forms a subgroup of $\mathbb{F}^{\times}$.
\end{problem}

\begin{solution}
	Let $G \subseteq \mathbb{F}^{\times}$ be the subset of non-zero elements represented by $\left < 1, d \right >$.

	Note that $1 = 1^{2} + d 0^{2} \in G$.

	Consider $x_{1}^{2} + d y_{1}^{2}, x_{2}^{2} + d y_{2}^{2} \in \mathbb{F}^{\times}$. It suffices to show that we can write
	\begin{align*}
		\frac{x_{1}^{2} + d y_{1}^{2}}{x_{2}^{2} + d y_{2}^{2}} & = \frac{(x_{1}^{2} + d y_{1}^{2}) (x_{2}^{2} - d y_{2}^{2})}{x_{2}^{4} - d y_{2}^{4}} \\
									& = \frac{x_{1}^{2} x_{2}^{2} - d x_{1}^{2} y_{2}^{2} + d x_{2}^{2} y_{1}^{2} - d y_{1}^{2} y_{2}^{2}}{x_{2}^{4} - d y_{2}^{4}}
	\end{align*}
\end{solution}

\begin{problem}[(6)]
	Let $\mathbb{E} / \mathbb{F}$ be a field extension of finite degree. Let $\ell : \mathbb{E} \to \mathbb{F}$ be a non-zero $\mathbb{F}$-linear map. Given any quasi-inner product $\beta$ on an $\mathbb{E}$-vector space $V$, show that the symmetric bilinear form $\ell \circ \beta$ on the underlying $\mathbb{F}$-vector space of $V$ is non-degenerate.
\end{problem}

\begin{solution}
	Suppose $v \in V$. Then there is some $w \in V$ such that $\beta(v, w) = a \neq 0$. Since $\ell$ is non-zero, there exists some $b \in \mathbb{E}$ with $\ell(b) \neq 0$. Thus, $\ell \circ \beta(v, bw/a) = \ell(b) \neq 0$.
\end{solution}
