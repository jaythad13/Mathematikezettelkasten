\begin{problem}[(1)]
	Prove \emph{Sylvester's theorem}: given a quadratic form $V, q$ over $\mathbb{R}$, with a decomposition $V, q \cong \left < 1 \right >^{\oplus r} \oplus \left < -1 \right >^{\oplus s}$, the integers $r, s$ are invariants of $V, q$. In particular, $r$ is the dimension of any maximal subspace of $V$ on which $q$ is positive definite.
\end{problem}

\pagebreak

\begin{problem}[(2)]
	Prove the \emph{Cartan--Dieudonn\'e theorem}. Let $V, q$ be a quadratic space of dimension $n$. Then any element of the orthogonal group is a product of at most $n$ reflections.
\end{problem}

\begin{solution}
	Hint: show $\mathrm{O}(V, q)$ acts transitively on the subsets $\{ v \in V \mid q(v) = a \}$ for each $a \in \mathbb{F}^{\times}$.
\end{solution}

\begin{problem}[(3)]
	Let $V, q$ be a quadratic space. Then the direct sum $V, q \oplus V, -q$ is hyperbolic --- isomorphic to the direct sum of copies of the hyperbolic plane $\left < 1, -1 \right >$
\end{problem}

\pagebreak

\begin{problem}[(4)]
	Let $\alpha, \beta \in \mathbb{F}^{\times}$ with $\alpha^{2} + \beta^{2} \neq 0$. Show that the quadratic form $V, q \cong \left < 1, -(\alpha^{2} + \beta^{2}) \right >$ has the property that the direct sum $V \oplus V$ is hyperbolic. Further, show that $V$ is hyperbolic if and only if $\alpha^{2} + \beta^{2}$ is a square in $\mathbb{F}$.
\end{problem}

\pagebreak

\begin{problem}[(5)]
	Suppose $d \in \mathbb{F}^{\times}$. Show that the collection of non-zero elements represented by the quadratic form $\left < 1, d \right >$ forms a subgroup of $\mathbb{F}^{\times}$.
\end{problem}

\pagebreak

\begin{problem}[(6)]
	Let $\mathbb{E} / \mathbb{F}$ be a field extension of finite degree. Let $\ell : \mathbb{E} \to \mathbb{F}$ be a non-zero $\mathbb{F}$-linear map. Given any quasi-inner product $\beta$ on an $\mathbb{E}$-vector space $V$, show that the symmetric bilinear form $\ell \circ \beta$ on the underlying $\mathbb{F}$-vector space of $V$ is non-degenerate.
\end{problem}
