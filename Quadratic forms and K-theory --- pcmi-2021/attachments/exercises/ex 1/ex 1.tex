\begin{problem}[(1)]
	Prove \emph{Sylvester's theorem}: given a quadratic form $V, q$ over $\mathbb{R}$, with a decomposition $V, q \cong \left < 1 \right >^{\oplus r} \oplus \left < -1 \right >^{\oplus s}$, the integers $r, s$ are invariants of $V, q$. In particular, $r$ is the dimension of any maximal subspace of $V$ on which $q$ is positive definite.
\end{problem}

\begin{solution}
	Suppose $\left < 1 \right >^{\oplus r_{1}} \oplus \left < -1 \right >^{\oplus s_{1}} \cong \left < 1 \right >^{\oplus r_{2}} \oplus \left < -1 \right >^{\oplus s_{2}}$ (given by an isomorphism $T : V_{1}, q_{1} \to V_{2}, q_{2}$). We claim that $r_{1} = r_{2}$ and $s_{1} = s_{2}$.

	Since the vector spaces must have equal dimension, $r_{1} + s_{1} = r_{2} + s_{2}$. Thus, it suffices to show $r_{1} = r_{2}$. We claim that the image of $U_{1} = \left < 1 \right >^{\oplus r_{1}}$ is contained in $\left < 1 \right >^{\oplus r_{2}}$. Else, $T^{\text{img}}(U_{1})$ would contain some subspace of $\left < -1 \right >^{\oplus s_{2}}$ on which $q_{2}$ is negative. This would force $q_{1}$ to be negative on $U_{1}$, which is false. 

	Suppose, without loss of generality, that $r_{1} \ge r_{2}$. Since $T$ is an isomorphism, $T_{\mid \left < 1 \right >^{\oplus r_{1}}} : \left < 1 \right >^{\oplus r_{1}} \to \left < 1 \right >^{\oplus r_{2}}$ is an injection from a higher-dimensional vector space to a lower-dimensional one, and thus, also an isomorphism. Thus, $r_{1} = r_{2}$.
\end{solution}

\begin{problem}[(2)]
	Let $V, q$ be a quadratic space of dimension $n$. Show that any element of the orthogonal group is a product of reflections.
\end{problem}

\begin{solution}
	We claim that the group generated by reflections and $(- \operatorname{id})$ acts transitively on ``spheres'' --- sets of the form $\{ v \in V \mid q(v) = a \}$ for $a \in \mathbb{F}^{\times}$. Suppose $v, w \in V$ have the same nonzero norm $q(v) = q(w) = a$ and let $u_{\pm} = v \pm w$. Note that $\beta(u_{+}, u_{-}) = 0$. Thus, $2 q(v) = q(u_{+}) + q(u_{-})$, and one of $u_{\pm}$ must be anisotropic. Using the fact that $2v = u_{+} + u_{-}$, we can show that $R_{u_{+}} v = - w$ if $u_{+}$ is anisotropic and $R_{u_{-}} v = w$ if $u_{-}$ is anisotropic. Even if $u_{-}$ is isotropic, we can compose with the reflection $R_{w}$ to get $R_{w}R_{u_{+}} v = w$.

	Now we induct on the dimension of $V$ to get the desired result. If $\dim V = 1$, then a quadratic linear isomorphism $T \in \mathrm{O}(V, q)$ is determined by a choice of $v \mapsto w$ for any non-zero $v, w \in V$ with $q(v) = q(w)$. If $v = w$ then $T = \operatorname{id}_{V}$ is the desired reflection, and else, $v - w$ is anisotropic, so $T = R_{u_{-}} : v \mapsto w$.

	Suppose $\dim V = n$. Diagonalise $V, q$ as $\left < a_{1}, \dots, a_{n - 1} \right > \oplus \left < a_{n} \right > = U \oplus \left < a_{n} \right >$ and let $e_{1}, \dots, e_{n}$ be the corresponding basis. For $T \in \mathrm{O}(V, q)$, let $R$ be the product of reflections with $R T e_{n} = e_{n}$. Since $RT$ maps $\left < a_{n} \right >$ to $\left < a_{n} \right >$ isomorphically, it must also map orthogonal complements $U$ into $U$ isomorphically. By induction, $(RT)_{\mid U} : U \to U$ is a product of reflections $S_{u_{i}} \in \mathrm{O}(U, q_{\mid U})$ through $u_{i} \in U$. The reflections $R_{u_{i}} \in \mathrm{O}(V, q)$ restrict to $R_{u_{i} \mid U} = S_{u_{i}}$ and have $R_{u_{i}} e_{n} = e_{n}$ (since $e_{n}$ is orthogonal to $U$). Thus, $RT$ is a product of all the $R_{u_{i}}$. Since $R^{-1}$ is a product of reflections, it follows that $T$ is a product of reflections.
\end{solution}

\begin{problem}[(3)]
	Let $V, q$ be a quadratic space. Show that the direct sum $(V, q) \oplus (V, -q)$ is hyperbolic --- isomorphic to the direct sum of copies of the hyperbolic plane $\left < 1, -1 \right >$.
\end{problem}

\begin{solution}
	Diagonalise to get $(V, q) \cong \left < a_{1}, \dots, a_{n} \right >$. Then $(V, q) \oplus (V, -q) \cong \left < a_{1}, -a_{1} \right > \oplus \dots \oplus \left < a_{n}, -a_{n} \right >$ by a permutation of basis.

	It suffices to give an isomorphism $\left < a, -a \right > \to \left < 1, -1 \right >$ for each $a \in \mathbb{F}^{\times}$. When $a$ is a square, we have an isomorphism $\left < a, -a \right > \to \left < 1, -1 \right >$ given by scaling by $\sqrt{a}$. 

	We can give a different isomorphism in the general case. Let $e_{1}, e_{2}$ be the basis corresponding to the diagonalisation  $\left < a, -a \right >$. Then consider $f_{1} = e_{1} + e_{2}$ and $f_{2} = e_{1} - e_{2}$. Then $q(f_{1}) = q(f_{2}) = 0$ and $\beta(f_{1}, f_{2}) = 2a \neq 0$. Thus, $V, q \cong H$. Explicitly, the diagonalisation  $V, q \cong \left < 1, -1 \right >$ is given by the basis $f_{1} + f_{2} / 4a, f_{1} - f_{2} / 4a$.
\end{solution}

\begin{problem}[(4)]
	Let $\alpha, \beta \in \mathbb{F}^{\times}$ with $\alpha^{2} + \beta^{2} \neq 0$. Show that the quadratic form $V, q \cong \left < 1, -(\alpha^{2} + \beta^{2}) \right >$ has the property that the direct sum $(V, q) \oplus (V, q)$ is hyperbolic. Further, show that $V$ is hyperbolic if and only if $\alpha^{2} + \beta^{2}$ is a square in $\mathbb{F}$.
\end{problem}

\begin{solution}
	Let $e_{1}, e_{2}, e_{3}, e_{4}$ be the basis corresponding to the diagonalisation
	\[
		(V, q) \oplus (V, q) \cong \left < 1, 1, -(\alpha^{2} + \beta^{2}), -(\alpha^{2} + \beta^{2}) \right >.
	\]
	For convenience, write $d = \alpha^{2} + \beta^{2}$.

	Consider $v_{1} = \alpha e_{1} + \beta e_{2} + e_{3}, v_{2} = \alpha e_{1} + \beta e_{2} + e_{4}$. They are both isotropic vectors with non-zero inner product $\beta_{q}(v_{1}, v_{2}) = \alpha^{2} + \beta^{2}$. Thus, $U = \operatorname{span}(v_{1}, v_{2})$ is hyperbolic.

	Now let $e_{1}, e_{2}$ be the basis of $V$ corresponding to the diagonalisation $V, q \cong \left < 1, -d \right >$. If $d = \gamma^{2} \in \mathbb{F}^{\times}$, then there is a quadratic isomorphism $V, q \to \left < 1, -1 \right >$ by $e_{1} \to e_{1}$ and $e_{2} \to \gamma e_{2}$. That is, if $\alpha^{2} + \beta^{2}$ is a square, then $V, q$ is hyperbolic. Since $q$ restricts to a quadratic form on $U$, we have $(V, q)^{\oplus 2} \cong (U, q_{\mid U}) \oplus (U^{\perp}, q_{\mid U^{\perp}})$. 

	To show $U^{\perp} \cong H$ also, it suffices to prove $U^{\perp}$ is isotropic (since every isotropic quadratic space contains $H$, every $2$-dimensional isotropic quadratic space is $H$). Consider $v \in V$, written in the basis $e_{1}, e_{2}, e_{3}, e_{4}$ as $(x, y, z, w)$. The $v \in U^{\perp}$ are exactly those satisfying the equations $\beta(v, v_{1}) = \beta(v, v_{2}) = 0$, or equivalently,
	\[
		w = z = \frac{\alpha x + \beta y}{d}.
	\]
	The isotropic $v$ are exactly those satisfying $x^{2} + y^{2} - d(z^{2} + w^{2}) = 0$. Substituting the expressions for $z, w$ in terms of $x, y$ from above, this equation is
	\[
		(\alpha^{2} - \beta^{2}) x^{2} + 4 \alpha \beta xy + (\beta^{2} - \alpha^{2}) y^{2} = 0.
	\]
	It suffices to provide a non-trivial solution to this equation. Choose $y = 1$. Then resulting quadratic polynomial in $x$ has discriminant
	\begin{align*}
		\Delta & = 16 \alpha^{2} \beta^{2} - 4 (\alpha^{2} - \beta^{2})(\beta^{2} - \alpha^{2}) \\
		       & = 4 (\alpha^{2} - \beta^{2})^{2} + 16 \alpha^{2} \beta^{2} \\
		       & = 4(\alpha^{2} + \beta^{2}).
	\end{align*}
	Since $\Delta$ is a square, the resulting quadratic equation in $x$ has a root in $\mathbb{F}$, which suffices to prove that $(V, q)^{\oplus 2} \cong (U, q_{\mid U}) \oplus (U^{\perp}, q_{\mid U^{\perp}}) \cong H^{\oplus 2}$.

	Conversely, if $\left < 1, -d \right >$ is hyperbolic, then it is isotropic. That is, there is a non-trivial solution $x^{2} - d y^{2} = 0$. Then $d = (x / y)^{2}$ is a square. That is, if $V, q$ is hyperbolic, then $-(\alpha^{2} + \beta^{2})$ is a square.
\end{solution}

\begin{problem}[(5)]
	Suppose $d \in \mathbb{F}^{\times}$. Show that the collection of non-zero elements represented by the quadratic form $\left < 1, d \right >$ forms a subgroup of $\mathbb{F}^{\times}$.
\end{problem}

\begin{solution}
	Let $G = \{ q(v) \neq 0 \mid v \in V \} \subseteq \mathbb{F}^{\times}$. We want to show $G \le \mathbb{F}^{\times}$.

	$1 \in G$ since $1 = 1^{2} + d 0^{2}$. 

	Suppose $x, y, z, w \in \mathbb{F}$ with $x^{2} + d y^{2}, z^{2} + d w^{2} \in \mathbb{F}^{\times}$. Then it's easy to verify that $(x^{2} + d y^{2})(z^{2} + d w^{2}) = (xz - dyw)^{2} + d (xw + yz)^{2}$ (it follows by the same trick as Diophantus' product of sum of squares identity. That is, $G$ is closed under products. This also allows us to show that $G$ is closed under taking inverses --- it leads us to try and solve the system $xz - dyw = 1, xw + yz = 0$. It is easy to verify that for $z = x / (x^{2} + d y^{2})$ and $w = y / (x^{2} + d y^{2})$ we have $(x^{2} + d y^{2}) (z^{2} + d w^{2}) = 1$.
\end{solution}

\begin{problem}[(6)]
	Let $\mathbb{E} / \mathbb{F}$ be a field extension of finite degree. Let $\ell : \mathbb{E} \to \mathbb{F}$ be a non-zero $\mathbb{F}$-linear map. Given any quasi-inner product $\beta$ on an $\mathbb{E}$-vector space $V$, show that the symmetric bilinear form $\ell \circ \beta$ on the underlying $\mathbb{F}$-vector space of $V$ is non-degenerate.
\end{problem}

\begin{solution}
	Suppose $v \in V$. Then there is some $w \in V$ such that $\beta(v, w) = a \neq 0$. Since $\ell$ is non-zero, there exists some $b \in \mathbb{E}$ with $\ell(b) \neq 0$. Thus, $\ell \circ \beta(v, bw/a) = \ell(b) \neq 0$.
\end{solution}
