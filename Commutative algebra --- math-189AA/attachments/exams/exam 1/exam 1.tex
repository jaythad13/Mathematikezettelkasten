Let $R$ be a commutative ring with unity.

\begin{problem}[1.]
	Let $I$ and $J$ be ideals in $R$ with $I \subseteq J$. Prove that there is a well-defined ring homomorphism from $R / I$ to $R / J$.
\end{problem}

\begin{solution}
	Let $\varphi : R \to R / J$ be the canonical surjection. Then we claim that $\overline{\varphi} : R / I \to R / J$ by $r + I \mapsto \varphi(r) + J$ is a well-defined ring homomorphism from $R / I$ to $R / J$. 

	First we show well-definedness. Suppose $r, s \in R$ and $r + I = s + I$. Then we have $r - s \in I \subseteq J$. Thus, $\varphi(r) - \varphi(s) = \varphi(r - s) = 0$. That is, $\varphi(r) = \varphi(s)$. It follows that
	\[
		\overline{\varphi}(r + I) = \varphi(r) + J = \varphi(s) + J = \overline{\varphi}(s + I).
	\]

	Now we show that $\overline{\varphi}$ is a homomorphism which it essentially inherits from $\varphi$. That is, for any $r, s \in R$ we have
	\begin{align*}
		\overline{\varphi}((r + I) + (s + I)) & = \overline{\varphi}((r + s) + I) \\
						      & = \varphi(r + s) + J \\
						      & = (\varphi(r) + \varphi(s)) + J \\
						      & = (\varphi(r) + J) + (\varphi(s) + J) \\
						      & = \overline{\varphi}(r + I) + \overline{\varphi}(s + I)
	\end{align*}
	and
	\begin{align*}
		\overline{\varphi}((r + I) (s + I)) & = \overline{\varphi}(rs + I) \\
						      & = \varphi(rs) + J \\
						      & = \varphi(r) \varphi(s) + J \\
						      & = (\varphi(r) + J) (\varphi(s) + J) \\
						      & = \overline{\varphi}(r + I) \overline{\varphi}(s + I).
	\end{align*}
	Thus, $\overline{\varphi}$ is a homomorphism.

\end{solution}

\begin{problem}[2.]
	Let $R = k[x_{1}, \dots, x_{n}]$ for some field $k$. Prove or disprove that $R / (x_{1}, x_{2}^{2}, \dots, x_{n}^{n})$ is local.
\end{problem}

\begin{solution}
	We prove that $R / (x_{1}, x_{2}^{2}, \dots, x_{n}^{n})$ is indeed local with unique maximal ideal $(\overline{x}_{1}, \dots, \overline{x}_{n})$. Let $I = (x_{1}, x_{2}^{2}, \dots, x_{n}^{n})$

	Recall that by the fourth isomorphism theorem, the ideals of $R / I$ are exactly $J / I$ for ideals $J \subseteq R$ with $I \subseteq J$. Note that this bijection is inclusion-preserving. Thus, it suffices to show that all ideals $J \subseteq R$ with $I \subseteq J$ are contained in a unique common maximal ideal $M$.

	Suppose $J \subseteq R$ and $I \subseteq J$. Then $J$ is contained in some maximal ideal $M_{J}$. Since $M_{J}$ is maximal, it is prime. Then,  $x_{i}^{i} \in I \subseteq J \subseteq M_{J}$ implies that $x_{i} \in M_{J}$ for each $i$. It follows that $M = (x_{1}, \dots, x_{n}) \subseteq M_{J}$. Since $R / M = k$ is a field, $M$ is maximal, and we have $M_{J} = M$. 

	That is, there is a unique common maximal ideal containing each $J$ containing $I$. $M / I = (\overline{x}_{1}, \dots, \overline{x}_{n})$ is then the unique maximal ideal of $R / I$ (here we write $x \mapsto \overline{x}$ under the canonical surjection $R \to R / I$).
\end{solution}

\begin{problem}[3.]
	Let $I \subseteq R$ be an ideal. Recall that the radical of an ideal is defined to be
	\[
		\sqrt{I} = \{ x \in R \mid x^{n} \in I \}
	\]
	\begin{enumerate}
		\item If $I$ is a proper ideal of $R$, show that $I = \sqrt{I}$ if and only if $I$ is the intersection of prime ideals.

		\item For a prime ideal $\mathfrak{p} \subseteq R$, show that $\sqrt{\mathfrak{p}^{n}} = \mathfrak{p}$ for any $n \in \mathbb{N}$.
	\end{enumerate}
\end{problem}

\begin{solution}
	\begin{enumerate}
		\item Note that $I \subseteq \sqrt{I}$ always since every $x \in I$ has $x^{1} \in I$. Thus, to show $I = \sqrt{I}$ it suffices to show $\sqrt{I} \subseteq I$.

			First suppose $I$ is the intersection of prime ideals --- $I = \bigcap_{i \in \mathscr{I}} \mathfrak{p}_{i}$. Suppose $x \in \sqrt{I}$. Then $x^{n} \in I$, and so $x^{n} \in \mathfrak{p}_{i}$ for each $i \in \mathscr{I}$. By primeness, we must have $x \in \mathfrak{p}_{i}$ for each $i \in \mathscr{I}$, and thus, $x \in I$. That is, $x \in \sqrt{I} \implies x \in I$ or $\sqrt{I} \subseteq I$.

			Now suppose $I = \sqrt{I}$. We claim that $\sqrt{I}$ is the intersection of all prime ideals $\mathfrak{p}$ containing $I$ and thus, $I$ is the intersection of prime ideals. In particular we show that $\sqrt{I} \subseteq \mathfrak{p}$ for every prime $\mathfrak{p} \supseteq I$ and $\bigcap_{\mathfrak{p} \supseteq I} \mathfrak{p} \subseteq \sqrt{I}$.

			Suppose $\mathfrak{p}$ is a prime ideal containing $I$. Suppose $x \in \sqrt{I}$ with $x^{n} \in I \subseteq \mathfrak{p}$. By primeness, we must have $x \in \mathfrak{p}$. That is, $\sqrt{I} \subseteq \mathfrak{p}$.

			Suppose $x \in \bigcap_{\mathfrak{p} \supseteq I} \mathfrak{p}$. Recall that there is a bijection between primes of $A$ containing $I$ and primes of $A / I$ given by pre-images and images under $A \to A / I$. Then, writing $A \to A / I$ by $a \mapsto \overline{a}$, we have $\overline{x} \in \mathfrak{N}_{A / I} = \bigcap_{\mathfrak{p} \subseteq A / I} \mathfrak{p}$. It follows that $\overline{x}$ is a nilpotent with $\overline{x}^{n} = 0$ for some $n \in  \mathbb{N}$ and so $x^{n} \in I$. Thus, $x \in \sqrt{I}$.

		\item We show double containment.

			Suppose $x \in \mathfrak{p}$. Then $x^{n} \in \mathfrak{p}^{n}$, and so $x \in \sqrt{\mathfrak{p}^{n}}$.

			Suppose $x \in \sqrt{\mathfrak{p}^{n}}$. Thus, $x^{m} \in \mathfrak{p}^{n}$ for some $m \in \mathbb{N}$. Writing $x^{m}$ as the product of $n$ elements of $\mathfrak{p}$, it follows that some $x^{\ell} \in \mathfrak{p}$. Thus, by primeness, $x \in \mathfrak{p}$.
	\end{enumerate}
\end{solution}

\begin{problem}[4.]
	A commutative ring is called reduced if its nilradical is the zero ideal. If $N$ is the nilradical of $R$, show that $R / N$ is reduced.
\end{problem}

\begin{solution}
	We show that $R / N$ has no non-zero nilpotents, and thus, is reduced.

	Suppose $x \in R / N$ and $x^{n} = 0$. Then $x^{n} \in N$. But since $N$ is the intersection of all prime ideals, we have $x^{n} \in \mathfrak{p}$ for all primes $\mathfrak{p} \subseteq R$. By primeness, this implies $x \in \mathfrak{p}$ for all primes of $R$, and thus, $x \in N$, which is their intersection.
\end{solution}

\begin{problem}[5.]
	Suppose $n > 1$. Prove that the following are equivalent:
	\begin{enumerate}
		\item $\mathbb{Z} / n \mathbb{Z}$ is a field.
		\item $\mathbb{Z} / n \mathbb{Z}$ is an integral domain.
		\item $n$ is a prime number.
	\end{enumerate}
\end{problem}

\begin{solution}
	Suppose $\mathbb{Z} / n \mathbb{Z}$ is a field. Then $(0)$ is maximal, and thus, prime. Thus, $\mathbb{Z} / n \mathbb{Z}$ is an integral domain.

	Suppose $\mathbb{Z} / n \mathbb{Z}$ is an integral domain. Then for any non-negative integers $a, b < n$ we have $ab \equiv 0$ only if $a = 0$ or $b = 0$. Since any divisors $a, b$ with $ab = n$ would have $ab \equiv 0$ this means that $n$ has no divisors apart from $1$ and itself. That is, $n$ is prime.

	Suppose $n$ is prime. Then, for all $a \not \equiv 0$ (that is all $a \neq n d$) we have $\gcd(a, n) = 1$. Then, by Bezout's theorem, there are some $b, q \in \mathbb{Z}$ such that $ab + nq = 1$. That is, $ab \equiv 1 \pmod{n}$. Thus, each non-zero $\overline{a} \in \mathbb{Z} / n \mathbb{Z}$ is a unit, with inverse $\overline{b}$. (Here we write $\mathbb{Z} \to \mathbb{Z} / n \mathbb{Z}$ by $a \mapsto \overline{a}$).

	We have shown $\text{(a)} \implies \text{(b)} \implies \text{(c)} \implies \text{(a)}$.
\end{solution}

\begin{problem}[6.]
	Let $N_{1} \subseteq N_{2} \subseteq \cdots$ be an ascending chain of $R$-submodules of $M$. Show that $\bigcup_{i = 1}^{\infty} N_{i}$ is a submodule of $M$.
\end{problem}

\begin{solution}
	We use the submodule criterion --- it suffices to show that $N = \bigcup_{i = 1}^{\infty} N_{i}$ is non-empty and that for each $x, y \in N$ and $r \in R$, we have $x - r y \in N$.

	Since $N_{1}$ is a submodule, it is non-empty. Since $N$ contains a non-empty set $N_{1}$, it is non-empty.

	Suppose $r \in R$, $x \in N_{i} \subseteq N$, and $y \in N_{j} \subseteq N$ and say $i \le j$ (without loss of generality) Then since $N_{i} \subseteq N_{j}$ we have $x, y \in N_{j}$. Since $N_{j}$ is a submodule, it satisfies the submodule criterion and $x - r y \in N_{j}$. Since $N_{j} \subseteq N$ we get $x - r y \in N$.
\end{solution}

\begin{problem}[7.]
	Let $R$ be a commutative domain. Show that
	\[
		R = \bigcap_{\mathfrak{p} \in \operatorname{Spec} R} R_{\mathfrak{p}}
	\]
	via the identification $x = x / 1$.
\end{problem}

\begin{solution}
	We consider the intersection of the isomorphic images of $R_{\mathfrak{p}}$ under their inclusion into the field of fractions $Q(R)$.

	Since $R \to R_{\mathfrak{p}}$ by $r \mapsto r / 1$ is an injection for each prime $\mathfrak{p}$, we have $R \subseteq R_{\mathfrak{p}}$ for each prime $\mathfrak{p}$. Thus, $R \subseteq \bigcap_{\mathfrak{p} \in \operatorname{Spec} R} R_{\mathfrak{p}}$.

	To show the converse we claim that the intersection of localisations away from multiplicative sets $R \setminus \mathfrak{p} = S_{\mathfrak{p}}$ is the localisation at the intersection of the multiplicative sets. This is true because $S_{\mathfrak{p}}^{-1} R = \{ r / s \mid r \in R, s \in S_{\mathfrak{p}} \}$ and $S_{\mathfrak{q}}^{-1} R = \{ r / s \mid r \in R, s \in S_{\mathfrak{q}} \}$ both quotiented by the same equivalence relation --- the equivalence relation of $Q(R)$. We remark without proof that this is well-defined since the intersection of multiplicative sets is a multiplicative set.

	Then it follows that
	\begin{align*}
		\bigcap_{\mathfrak{p} \in \operatorname{Spec} R} R_{\mathfrak{p}} & = \left ( \bigcap_{\mathfrak{p} \in \operatorname{Spec} R} S_{\mathfrak{p}} \right )^{-1} R \\
										  & = \left ( \bigcap_{\mathfrak{p} \in \operatorname{Spec} R} R \setminus \mathfrak{p} \right )^{-1} R \\
										  & = \left ( R \setminus \bigcup_{\mathfrak{p} \in \operatorname{Spec} R} \mathfrak{p} \right )^{-1} R.
	\end{align*}

	We claim that $\bigcup_{\mathfrak{p} \in \operatorname{Spec} R} \mathfrak{p}$ is exactly the non-units in $R$. Clearly everything in the union of prime ideals is not a unit (else the ideals would be $R$, and thus, not prime). Conversely, every non-unit is contained in a maximal, and thus, prime ideal. Thus,
	\begin{align*}
		\bigcap_{\mathfrak{p} \in \operatorname{Spec} R} R_{\mathfrak{p}} & = \left ( R \setminus \bigcup_{\mathfrak{p} \in \operatorname{Spec} R} \mathfrak{p} \right )^{-1} R \\
										  & = \{ \text{units} \}^{-1} R \\
										  & = R.
	\end{align*}

	Here the last equality follows because localising away from the set of units does not change the ring. In particular, for any unit $u \in R$ with inverse $u^{-1} \in R$, we have $1 / u \sim u^{-1}$ since $u(1 /u - u^{-1}) = 1 - 1 = 0$. Thus, the localisation contains no new elements and the injection $R \to \{ \text{units} \}^{-1} R$ is surjective.
\end{solution}
