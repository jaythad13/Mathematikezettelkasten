Let $R$ be a commutative ring with unity.

\begin{problem}[1.]
	Let $I$ and $J$ be ideals in $R$ with $I \subseteq J$. Prove that there is a well-defined ring homomorphism from $R / I$ to $R / J$.
\end{problem}

\pagebreak

\begin{problem}[2.]
	Let $R = k[x_{1}, \dots, x_{n}]$ for some field $k$. Prove or disprove that $R / (x_{1}, x_{2}^{2}, \dots, x_{n}^{2})$ is local.
\end{problem}

\pagebreak

\begin{problem}[3.]
	Let $I \subseteq R$ be an ideal. Recall that the radical of an ideal is defined to be
	\[
		\sqrt{I} = \{ x \in R \mid x^{n} \in I \}
	\]
	\begin{enumerate}
		\item If $I$ is a proper ideal of $R$, show that $I = \sqrt{I}$ if and only if $I$ is the intersection of prime ideals.

		\item For a prime ideal $\mathfrak{p} \subseteq R$, show that $\sqrt{\mathfrak{p}^{n}} = \mathfrak{p}$ for any $n \in \mathbb{N}$.
	\end{enumerate}
\end{problem}

\pagebreak

\begin{problem}[4.]
	A commutative ring is called reduced if its nilradical is the zero ideal. If $N$ is the nilradical of $R$, show that $R / N$ is reduced.
\end{problem}

\pagebreak

\begin{problem}[5.]
	Suppose $n > 1$. Prove that the following are equivalent:
	\begin{enumerate}
		\item $\mathbb{Z} / n \mathbb{Z}$ is a field.
		\item $\mathbb{Z} / n \mathbb{Z}$ is an integral domain.
		\item $n$ is a prime number.
	\end{enumerate}
\end{problem}

\pagebreak

\begin{problem}[6.]
	Let $N_{1} \subseteq N_{2} \subseteq \cdots$ be an ascending chain of $R$-submodules of $M$. Show that $\bigcup_{i = 1}^{\infty} N_{i}$ is a submodule of $M$.
\end{problem}

\pagebreak

\begin{problem}[7.]
	Let $R$ be a commutative domain. Show that
	\[
	R = \bigcap_{\mathfrak{p} \in \operatorname{Spec} R} R_{\mathfrak{p}}
	\]
	via the identification $x = x / 1$.
\end{problem}
