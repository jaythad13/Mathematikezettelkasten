\begin{problem}[2.]
	Let $f: A \to \mathbb{R}$ be integrable and let $g = f$ except at finitely many points. Show that $g$ is integrable and $\int_{A} f = \int_{A} g $.
\end{problem}

\pagebreak

\begin{problem}[5.]
	Let $f, g: A \to \mathbb{R}$ be integrable and suppose $f \le g$. Show that $\int_{A} f \le \int_{A} g$.
\end{problem}

\pagebreak

\begin{problem}[8.]
	Prove that $[a_{1}, b_{1}] \times \dots \times [a_{n}, b_{n}]$ does not have content $0$ if $a_{i} < b_{i}$ for each $i$.
\end{problem}

\pagebreak

\begin{problem}[11]
	Let $A$ be the set of Problem 1-18. If $\sum_{i = 1}^{\infty} (b_{i} - a_{i}) < 1$, show that the boundary of $A$ does not have measure $0$.
\end{problem}

\pagebreak

\begin{problem}[13.]
	\begin{enumerate}
		\item Show that the collection of all rectangles $[a_{1}, b_{1}] \times \dots \times [a_{n}, b_{n}]$ with all $a_{i}$ and $b_{i}$ rational can be arranged in a sequence.
		\item If $A \subset \mathbb{R}^{n}$ is any set and $\mathcal{O}$ is an open cover of $A$, show that there is a sequence $U_{1}, U_{2}, U_{3}, \dots$ of members of $\mathcal{O}$ which also cover $A$.
	\end{enumerate}
\end{problem}

\pagebreak

\begin{problem}[14.]
	Show that if $f, g : A \to \mathbb{R}$ are integrable, so is $f g$.
\end{problem}

\pagebreak

\begin{problem}[15.]
	Give an example of a bounded set of measure $0$ such that $\int_{A} \chi_{C}$ does not exist.
\end{problem}

\pagebreak

\begin{problem}[17.]
	If $C$ is a bounded set of measure $0$ and $\int_{A} \chi_{C} = 0$.
\end{problem}

\pagebreak

\begin{problem}[19.]
	Let $U$ be the open set of Problem 3-11. Show that if $f = \chi_{U}$ except on a set of measure $0$, then $f$ is not integrable on $[0, 1]$.
\end{problem}

\pagebreak

\begin{problem}[22.]
	If $A$ is a Jordan-measurable set and $\varepsilon > 0$, show that there is a compact Jordan-measurable set $C \subset A$ such that $\int_{A \setminus C} 1$.
\end{problem}
