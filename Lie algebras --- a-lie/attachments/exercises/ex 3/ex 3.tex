\begin{problem}[3.]
	Consider the matrix Lie algebra (with bracket given by the commutator)
	\[
		L = \left \{ \begin{pmatrix}
				0 & -a_{3} & a_{2} \\
				a_{3} & 0 & -a_{1} \\
				-a_{2} & a_{1} & 0
		\end{pmatrix} \mid a_{1}, a_{2}, a_{3} \in \mathbb{R} \right \}.
	\]
	Show that
	\begin{align*}
		\varphi & : (\mathbb{R}^{3}, \times) \to L \\
			& : \begin{pmatrix}
				a_{1} \\
				a_{2} \\
				a_{3}
			\end{pmatrix} \mapsto \begin{pmatrix}
				0 & -a_{3} & a_{2} \\
				a_{3} & 0 & -a_{1} \\
				-a_{2} & a_{1} & 0
		\end{pmatrix}
	\end{align*}
	is a Lie algebra isomorphism.
\end{problem}
