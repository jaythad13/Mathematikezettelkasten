\begin{problem}[1]
	Show that for any $x, y \in \mathfrak{gl}(V)$
	\[
		(\operatorname{ad}_{x})^{n} y = \sum_{k = 0}^{n} (-1)^{k} \binom{n}{k} x^{n - k} y x^{k}.
	\]
\end{problem}

\pagebreak

\begin{problem}[2]
	Suppose $L \subset \mathfrak{gl}(V)$ is a Lie algebra with subalgebra $H$. Let $\mu : H \to \mathbb{F}$ be a (not necessarily linear) function to the base field such that
	\[
		\{ v \in V \mid hv = \mu(h) v \text{ for all } h \in H \} \neq 0.
	\]
	Show that $\mu$ is a linear map.
\end{problem}
