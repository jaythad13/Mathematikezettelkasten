\begin{problem}[2.]
	We say that a Lie algebra $L$ is \emph{abelian} if $[x, y] = 0$ for all $x, y \in L$. Show that every $1$-dimensional Lie algebra is abelian.
\end{problem}

\begin{solution}
	Suppose $L$ is one-dimensional and spanned by $v \in L$. Then for any $x, y \in L$, we have $x = \alpha v$, $y = \beta v$, where $\alpha, \beta$ are scalars. Thus,
	\[
		[x, y] = \alpha \beta [v, v] = 0.
	\]
	That is, $L$ is abelian.
\end{solution}

\begin{problem}[3.]
	Let $E_{i j}$ be the matrix with all zeroes except a $1$ in the $i$th row, $j$th column. Show
	\[
		[E_{i j}, E_{k l}] = \delta_{j k} E_{i l} - \delta_{i l} E_{k j}.
	\]
\end{problem}

\begin{problem}[4.]
	The \emph{symplectic group} is
	\[
		\operatorname{SP}_{4}(\mathbb{C}) = \{ A \in M_{4 \times 4} \mid A^{\mathrm{T}} J A \}
	\]
	for
	\[
		J = \begin{pmatrix}
			0 & I_{2} \\
			I_{2} & 0
		\end{pmatrix}.
	\]

	Show that its Lie algebra is
	\[
		\mathfrak{sp}_{4}(\mathbb{C}) = \{ x \in M_{4 \times 4} \mid J x - x^{\mathrm{T}} J = 0 \}.
	\]
\end{problem}
