\begin{problem}[1]
	Show $\mathfrak{n}_{3}(F)$, the set of strictly upper triangular $3$-by-$3$ matrices is a nilpotent Lie algebra.
\end{problem}

\pagebreak

\begin{problem}[2]
	Show $(L^{(m)})^{(n)} = L^{(m + n)}$.
\end{problem}

\pagebreak

\begin{problem}[3]
	Show every nilpotent Lie algebra is soluble and give an example of a soluble Lie algebra that is not nilpotent.
\end{problem}

\pagebreak

\begin{problem}[4]
	Show
	\[
		L = \left \{ \begin{pmatrix}
				a & b & 0 & 0 \\
				c & -a & 0 & 0 \\
				0 & 0 & x & y \\
				0 & 0 & z & -x
		\end{pmatrix} \mid a, b, c, x, y, z \in F \right \}
	\]
	is semisimple, but not simple.
\end{problem}
