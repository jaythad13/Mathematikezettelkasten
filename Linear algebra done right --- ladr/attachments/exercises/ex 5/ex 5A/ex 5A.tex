\begin{problem}[1]
	Suppose $T \in \mathcal{L}(V)$ and $U$ is a subspace of $V$.
	\begin{enumerate}
		\item Prove that if $U \subset \operatorname{null} T$, then $U$ is invariant under $T$.
		\item Prove that if $\operatorname{range} T \subset U$, then $U$ is invariant under $T$.
	\end{enumerate}
\end{problem}

\pagebreak

\begin{problem}[2]
	Suppose that $T \in \mathcal{L}(V)$ and $V_{1}, \dots, V_{m}$ are subspaces of $V$ invariant under $T$. Prove that $V_{1} + \dots + V_{m}$ is invariant under $T$.
\end{problem}

\pagebreak

\begin{problem}[3]
	Suppose $T \in \mathcal{L}(V)$. Prove that the intersection of every collection of subspaces of $V$ invariant under $T$ is invariant under $T$.
\end{problem}

\pagebreak

\begin{problem}[4]
	Prove or give a counterexample: if $V$ is finite-dimensional and $U$ is a subspace of $V$ that is invariant under every operator on $V$, then $U = \{ 0 \}$ or $U = V$.
\end{problem}

\pagebreak

\begin{problem}[5]
	Suppose $T \in \mathcal{L}(\mathbb{R}^{2})$ is defined by $T(x, y) = (-3y, x)$. Find the eigenvalues of $T$.
\end{problem}

\pagebreak

\begin{problem}[6]
	Define $T \in \mathcal{L}(\mathbb{F}^{2})$ by $T(w, z) = (z, w)$. Find all the eigenvalues and eigenvectors of $T$.
\end{problem}

\pagebreak

\begin{problem}[7]
	Define $T \in \mathcal{L}(\mathbb{F}^{3})$ by $T(z_{1}, z_{2}, z_{3}) = (2z_{2}, 0, 5z_{3})$. Find all eigenvalues and eigenvectors of $T$.
\end{problem}

\pagebreak

\begin{problem}[8]
	Suppose $P \in \mathcal{L}(V)$ is such that $P^{2} = P$. Prove that if $\lambda$ is an eigenvalue of $P$ then $\lambda = 0$ or $\lambda = 1$.
\end{problem}
