\begin{problem}[1.2.21]
	Show that functors preserve isomorphism. That is, prove that if $F : \mathscr{A} \to \mathscr{B}$ is a functor, and $A, A' \in \mathscr{A}$ with $A \cong A'$, then $F(A) \cong F(A')$.
\end{problem}

\begin{solution}
	If $A \cong A'$, then there exist morphisms $f : A \to A'$ and $f' : A' \to A$ with $f \circ f' = \operatorname{id}_{A}$ and $f' \circ f = \operatorname{id}_{A'}$. We claim $F(f)$ and $F(f')$ serve the same purpose for $F(A)$ and $F(A')$.

	In particular, $F(f) \circ F(f') : F(A) \to F(A)$ is just
	\[
		F(f \circ f') = F(\operatorname{id}_{A}) = \operatorname{id}_{F(A)}.
	\]
	Similarly, $F(f') \circ F(f) : F(A') \to F(A')$ is just 
	\[
		F(f' \circ f) = F(\operatorname{id}_{A'}) = \operatorname{id}_{F(A')}.
	\]
	Thus, $F(f) : F(A) \to F(A')$ is the desired isomorphism.
\end{solution}

\begin{problem}[1.2.22]
	Prove the assertion made in Example 1.2.9. In other words, given ordered sets $A$ and $B$, and denoting by $\mathscr{A}$ and $\mathscr{B}$ the corresponding categories, show that a functor $\mathscr{A} \to \mathscr{B}$ amounts to an order-preserving map $A \to B$.
\end{problem}

\begin{solution}
	In a pre-order category $\mathscr{P}$ corresponding to the pre-ordered set $P$, there exists a morphism $f : p_{1} \to p_{2}$ if and only if $p_{1} \le p_{2}$ in $P$. A functor $F : \mathscr{A} \to \mathscr{B}$ has a map $F(a_{1}) \to F(a_{2})$ if and only if there is a map $a_{1} \to a_{2}$. Equivalently, on the corresponding pre-ordered sets, $F(a_{1}) \le F(a_{2})$ in $B$ if and only if $a_{1} \le a_{2}$ in $A$ --- $F$ is an order preserving map.
\end{solution}

\begin{problem}[1.2.25]
	Sometimes we meet functors whose domain is a product $\mathscr{A} \times \mathscr{B}$ of categories. Here you will show that such a functor can be regarded as an interlocking pair of families of functors, one defined on $\mathscr{A}$ and the other defined on $\mathscr{B}$. (This is very like the situation for bilinear and linear maps).
	\begin{enumerate}
		\item Let $F : \mathscr{A} \times \mathscr{B} \to \mathscr{C}$ be a functor. Prove that for each $A \in \mathscr{A}$, there is a functor $F^{A} : \mathscr{B} \to \mathscr{C}$ defined on objects $B \in \mathscr{B}$ by $F^{A}(B) = F(A, B)$ and on maps $g \in \mathscr{B}$ by $F^{A}(g) = F(\operatorname{id}_{A}, g)$. Prove that for each $B \in \mathscr{B}$, there is a functor $F_{B} : \mathscr{A} \to \mathscr{C}$ defined similarly.
		\item Let $F : \mathscr{A} \times \mathscr{B} \to \mathscr{C}$ be a functor. With notation as in (a), show that the families of functors $(F^{A})_{A \in \mathscr{A}}$ and $(F_{B})_{B \in \mathscr{B}}$ satisfy the following two conditions:
			\begin{enumerate}[label = (\roman*)]
				\item if $A \in \mathscr{A}$ and $B \in \mathscr{B}$, then $F^{A}(B) = F_{B}(A)$ 
				\item $f : A \to A'$ in $\mathscr{A}$ and $g : B \to B'$ in $\mathscr{B}$, then $F^{A'}(g) \circ F_{B}(f) = F_{B'}(f) \circ F^{A}(g)$.
			\end{enumerate}
		\item Now take categories $\mathscr{A}$, $\mathscr{B}$, and $\mathscr{C}$, and take families of functors $(F^{A})_{A \in \mathscr{A}}$ and $(F_{B})_{B \in \mathscr{B}}$ satisfying the two conditions in (b). Prove that there is a unique functor $F : \mathscr{A} \times \mathscr{B} \to \mathscr{C}$ satisfying the equations in (a).
	\end{enumerate}
\end{problem}
