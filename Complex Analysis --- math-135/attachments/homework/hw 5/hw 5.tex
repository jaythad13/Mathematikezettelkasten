\begin{problem}[14.]
	Prove that all entire functions that are also injective take the form $f(z) = az + b$ with $a, b \in \mathbb{C}$ and $a \neq 0$.
\end{problem}

\begin{solution}
	Suppose $f$ is entire and injective. Then by the chain rule, $g : z \mapsto f(1/z)$ is holomorphic on $\mathbb{C} \setminus \{ 0 \}$, and since $1/z$ is injective, so is $g$. Using the power series expansion for $f$, we can write a Laurent series expansion for $g$ on all of $\mathbb{C} \setminus \{ 0 \}$ --- if $f(z) = \sum_{n = 0}^{\infty} a_{n} z^{n}$, then
	\[
		g(z) = \sum_{n = 0}^{\infty} \frac{a_{n}}{z^{n}} = \sum_{n = -\infty}^{0} a_{-n} z^{n}.
	\]

	We claim that $g$ can't have infinitely many non-zero coefficients --- $g$ cannot have an essential singularity at $0$. Suppose, by way of contradiction, that $g$ did. Then by Casorati-Weierstrass, the image $g^{\text{img}}(\mathbb{D} \setminus \{ 0 \})$ would be dense in $\mathbb{C}$. Consider any open set $U \in \mathbb{C} \setminus \mathbb{D}$. By the open mapping theorem $g^{\text{img}}(U)$ is open in in $\mathbb{C}$. Note that the intersection of an open set and a dense set $g^{\text{img}}(U) \cap g^{\text{img}}(\mathbb{D} \setminus \{ 0 \})$ is non-empty. Since $U \cap \mathbb{D}$ is empty, $g$ must be non-injective on $\mathbb{C} \setminus \{ 0 \}$, contradicting our earlier observation that $g$ is injective as the composition of two injective functions.

	If $g$ has only finitely many coefficients (suppose all $a_{-m}$ with $m > n$ are zero) terms in its Laurent series expansion, then $f$ must be a polynomial $a_{0} + a_{1} z + \dots + a_{n} z^{n}$. Suppose by way of contradiction that $n > 1$. Then, by the fundamental theorem of algebra, $f$ has $n$ roots. If any of them are distinct, then $f$ cannot be injective and we have a contradiction. If $f$ has $n$ repeated roots, we have $f(z) = (z - z_{0})^{n}$. Since $(z - z_{0})^{n} = 1$ has $n$ distinct solutions, $f$ is not injective here either, and again, we have a contradiction. Thus $n \le 1$. Of course, if $n < 1$ $f$ is just constant and not injective, so we must have $n = 1$. That is, $f = az + b$ with $a \neq 0$.
\end{solution}

\begin{problem}[15.]
	Use the Cauchy inequalities or the maximum modulus principle to solve the following problems:
	\begin{enumerate}
		\item Prove that if $f$ is an entire function that satisfies
			\[
				\sup_{\lvert z \rvert = R} \lvert f(z) \rvert \le A R^{k} + B
			\]
			for all $R > 0$, and for some integer $k \ge 0$, and some constants $A, B > 0$, then $f$ is a polynomial of degree $\le k$.
		\setcounter{enumi}{3}
		\item Show that if the real part of an entire function $f$ is bounded, then $f$ is constant.
	\end{enumerate}
\end{problem}

\begin{solution}
	\begin{enumerate}
		\item Since $f$ is entire it has a power series expansion
			\[
				f(z) = \sum_{n = 0}^{\infty} a_{n} z^{n}
			\]
			about $0$ that converges on all of $\mathbb{C}$. We claim that $a_{n} = 0$ for all $n > k$, and thus, $f$ is a polynomial.

			By the Cauchy inequalities and the fact that $f$ is entire we have
			\[
				\lvert f^{(n)}(z) \rvert \le \frac{n! \sup_{\lvert z \rvert = R} \lvert f(z) \rvert}{R^{n}} \le \frac{n! (AR^{k} + B)}{R^{n}}
			\]
			for all $R > 0$. For $n > k$, as $R \to \infty$, this bound converges to $0$. Thus, we must have $f^{(n)}(z) = 0$ for all $n > k$.
		\setcounter{enumi}{3}
		\item Consider $g : z \to f(1/z)$. $g$ cannot have an essential singularity at $0$ --- since $\operatorname{Re} f$ is bounded, the image of the punctured disc around $0$ under $g$ is not dense in $\mathbb{C}$. Thus, by a similar Laurent series argument to that in problem 1, $f$ is a polynomial. Non-constant polynomials must have unbounded real part, so $f$ must be constant.
	\end{enumerate}
\end{solution}

\begin{problem}[16.]
	Suppose $f$ and $g$ are holomorphic in a region containing the disc $\lvert z \rvert \le 1$. Suppose that $f$ has a simple zero at $z = 0$ and vanishes nowhere else in $\lvert z \rvert \le 1$.

	Let
	\[
		f_{\varepsilon}(z) = f(z) + \varepsilon g(z).
	\]
	Show that if $\varepsilon$ is sufficiently small, then
	\begin{enumerate}
		\item $f_{\varepsilon}(z)$ has a unique zero in $\lvert z \rvert \le 1$, and
		\item if $z_{\varepsilon}$ is this zero, the mapping $\varepsilon \mapsto z_{\varepsilon}$ is continuous.
	\end{enumerate}
\end{problem}

\begin{solution}
	\begin{enumerate}
		\item Note that since $g$ is holomorphic in a region containing the disc $\overline{\mathbb{D}}$, it must be bounded on the compact circle $\lvert z \rvert = 1$. Thus, for sufficiently small $\varepsilon > 0$, we have $\lvert f \rvert \ge \lvert \varepsilon g \rvert$ everywhere on the circle. Then by Rouche's theorem, $f$ and $f + \varepsilon g$ have the same zeroes inside $\mathbb{D}$ --- that is, $f + \varepsilon g$ has one simple zero in $\mathbb{D}$.
		\item Suppose $\varepsilon < \delta$ is sufficient to ensure that $z_{\varepsilon}$ is a zero of $f_{\varepsilon}$. Consider the map $(0, \delta) \to \mathbb{D}$ by $\varepsilon \mapsto z_{\varepsilon}$ where $z_{\varepsilon} = x_{\varepsilon} + i y_{\varepsilon}$ is the unique zero of $f_{\varepsilon}$ in the disc. The map $\varepsilon \mapsto (x_{\varepsilon}, y_{\varepsilon})$ is then the implicit function $h$ that gives  $F(h(\varepsilon), \varepsilon) = 0$ for $F : ((x, y), \varepsilon) \to f_{\varepsilon}(x + i y)$.

			Since $F(0, 0) = f(0) = 0$, by the implicit function theorem we have that $h$ is differentiable, and thus, continuous as long as $f'(0) \neq 0$. This is true since $f(0)$ is a simple zero. Thus $h : \varepsilon \to z_{\varepsilon}$ is continuous.
	\end{enumerate}
\end{solution}

\begin{problem}[18.]
	Give another proof of the Cauchy integral formula
	\[
		f(z) = \frac{1}{2 \pi i} \int_{C} \frac{f(\zeta)}{\zeta - z} \, d\zeta
	\]
	using homotopy of curves.
\end{problem}

\begin{solution}
	Suppose $f$ is holomorphic on the circle $C$ and the disc $D$ that it bounds. Then the quotient $\frac{f(\zeta)}{\zeta - z}$ is holomorphic everywhere on $C$ and $D$ except $z$. We claim that using homotopy, we can show
	\[
		\int_{C} \frac{f(\zeta)}{\zeta - z} \, d\zeta = \int_{C_{\varepsilon}(z)} \frac{f(\zeta)}{\zeta - z} \, d\zeta.
	\]

	Consider the curve $\Gamma$ which is the union of $C_{\varepsilon}(z)$, the circle of radius $\varepsilon$ centred at $z$, and a line connecting it to $C$. Clearly $\Gamma$ and $C$ are homotopic. Further, the integral over $\Gamma$ is just the integral over $C_{\varepsilon}(z)$ since we integrate over the straight line in opposite directions, cancelling the integral out. Thus, we have the result above.

\begin{figure}[H]
    \centering
    \incfig{drawing}
    \caption{$\Gamma$ and $C$.}
    \label{fig:parking-procedure}
\end{figure}

	Now we can just imitate the standard proof of the Cauchy integral formula. We rewrite
	\[
		\int_{C_{\varepsilon}(z)} \frac{f(\zeta)}{\zeta - z} \, d\zeta = \int_{C_{\varepsilon}(z)} \frac{f(\zeta) - f(z)}{\zeta - z} \, d\zeta + f(z) \int_{C_{\varepsilon}(z)} \frac{1}{\zeta - z} \, d\zeta 
	\]
	Since $f$ is holomorphic, the first integrand, the difference quotient, is bounded, say by $M$, and thus
	\[
		\left \lvert \int_{C_{\varepsilon}(z)} \frac{f(\zeta) - f(z)}{\zeta - z} \, d \zeta  \right \rvert \le M 2 \pi \varepsilon
	\]
	Thus, the integral goes to zero as $\varepsilon \to 0$. The second integral, we've already verified is $2 \pi i$, and thus,
	\[
		\int_{C} \frac{f(\zeta)}{\zeta - z} \, d\zeta = \int_{C_{\varepsilon}(z)} \frac{f(\zeta)}{\zeta - z} \, d\zeta = 2 \pi i f(z)
	\]
	which is just the Cauchy integral formula, as desired.
\end{solution}

\begin{problem}[19.]
	Prove the maximum principle for harmonic functions:
	\begin{enumerate}
		\item If $u$ is a non-constant real-valued harmonic function in a region $\Omega$, then $u$ cannot attain a maximum (or a minimum) in $\Omega$.
		\item Suppose that $\Omega$ is a region with compact closure $\overline{\Omega}$. If $u$ is harmonic in $\Omega$ and continuous in $\overline{\Omega}$, then
			\[
				\sup_{z \in \Omega} \lvert u(z) \rvert \le \sup_{z \in \overline{\Omega} \setminus \Omega} \lvert u(z) \rvert
			\]
	\end{enumerate}
\end{problem}

\begin{solution}
	\begin{enumerate}
		\item Recall that if $u$ is a non-constant real harmonic function on $\Omega$, then we can write $u = \operatorname{Re} f$ for some non-constant holomorphic function $\Omega \to \mathbb{C}$. By the open mapping theorem, $f^{\text{img}}(\Omega)$ is open. Thus, for any $z_{0} \in \Omega$, there exist some $z_{1}, z_{2} \in \Omega$ with $\operatorname{Re} f(z_{1}) > \operatorname{Re} f(z_{0})$ and $\operatorname{Re} f(z_{2}) < \operatorname{Re} f(z_{0})$. 

			In particular, since there is some $r > 0$ with $N_{r}(f(z)) \subset f^{\text{img}}(\Omega)$, choose $f(z_{1}) = f(z_{0}) + \delta \in N_{r}(f(z))$ and $f(z_{2}) = f(z_{0}) - \delta \in N_{r}(f(z))$ such that $0 < \delta < r$. That is, $u(z_{2}) < u(z_{0}) < u(z_{1})$, and thus, $u$ cannot have a maximum or minimum on $\Omega$.
		\item If $u$ is harmonic on a region $\Omega$ with compact closure $\overline{\Omega}$, $\lvert u \rvert$ must attain a maximum on $\overline{\Omega}$ (since it is a continuous function on a compact set). But a maximum of $\lvert u \rvert$ is a maximum or a minimum of $u$, and thus, cannot occur on $\Omega$ unless $u$ is constant.

			In either case, $\lvert u \rvert^{\text{img}}(\overline{\Omega} \setminus \Omega)$ contains an upper bound on $\lvert u \rvert^{\text{img}}(\Omega)$, and thus,
			\[
				\sup_{z \in \Omega} \lvert u(z) \rvert \le \sup_{z \in \overline{\Omega} \setminus \Omega} \lvert u(z) \rvert.
			\]
	\end{enumerate}
\end{solution}
