\begin{problem}[10. (a)]
	Find the Hadamard product for $e^{z} - 1$.
\end{problem}

\begin{solution}
	Note that $e^{z} - 1 = 0$ exactly at $z = a_{n} = 2 n \pi i$ for all integers $n$. Also, $e^{z} - 1$ clearly has order of growth $1$. It has order of growth at most $1$ since
	\[
		\lvert e^{z} - 1 \rvert \le \lvert e^{z} \rvert + 1 \le 2 \lvert e^{\lvert z \rvert} \rvert
	\]
	and it has order of growth at least $1$, since
	\[
		\lvert e^{z} - 1 \rvert \ge \lvert e^{z} \rvert - 1
	\]
	which is eventually greater than any $A e^{B \lvert z \rvert^{1 - \varepsilon}}$.

	Thus,
	\begin{align*}
		e^{z} - 1 & = e^{az + b} z \prod_{n = 1}^{\infty} E_{1} \left ( \frac{z}{2n \pi i} \right ) E_{1} \left ( \frac{z}{- 2n \pi i} \right ) \\
			  & = e^{az + b} z \prod_{n = 1}^{\infty} e^{z / 2 n \pi i} \left ( 1 - \frac{z}{2n \pi i} \right )  e^{- z / 2n \pi i} \left ( 1 + \frac{z}{2n \pi i} \right ) \\
			  & = e^{az + b} z \prod_{n = 1}^{\infty} \left ( 1 + \frac{z^{2}}{4 n^{2} \pi^{2}} \right ).
	\end{align*}

	We can solve for $a, b \in \mathbb{C}$ by calculating the infinite product. Compare the product to
	\[
		\sin{z} = z \prod_{n = 1}^{\infty} \left ( 1 - \frac{z^{2}}{\pi^{2} n^{2}} \right )
	\]
	and notice that
	\[
		\frac{z}{2i} \prod_{n = 1}^{\infty} \left ( 1 + \frac{z^{2}}{4 \pi^{2} n^{2}} \right ) = \sin{\left ( \frac{z}{2 i} \right )}.
	\]
	Using Euler's formula, we get
	\[
		z \prod_{n = 1}^{\infty} \left ( 1 + \frac{z^{2}}{4 \pi^{2} n^{2}} \right ) = 2i \frac{e^{iz / 2i} - e^{-iz / 2i}}{2i} = e^{z / 2} - e^{-z / 2}.
	\]

	Now we can solve for $a$ and $b$. Just by observation, $a = 1/2$ and $b = 0$ gives us the Hadamard factorisation
	\begin{align*}
		e^{z/2} z \prod_{n = 1}^{\infty} \left ( 1 + \frac{z^{2}}{4 n^{2} \pi^{2}} \right ) & = e^{z/2} (e^{z/2} - e^{-z/2}) \\
												       & = e^{z} - 1.
	\end{align*}
\end{solution}

\begin{problem}[11.]
	Show that if  $f$ is an entire function of finite order that omits two values, then $f$ is constant. This result remains true for any entire function and is known as Picard's little theorem.
\end{problem}

\begin{solution} 
	Without loss of generality, we can assume that $f$ has no zeroes, since it is only a translation away from a function that has no zeroes. Thus, $f$ is an entire, non-vanishing function of finite order, and must have Hadamard factorisation
	\[
		f(z) = e^{p(z)}
	\]
	for some polynomial $p$. Non-constant polynomials are surjective on $\mathbb{C}$ by the fundamental theorem of calculus, and the complex plane $\mathbb{C}$ has image $\mathbb{C} \setminus \{ 0 \}$ under the exponential function $z \mapsto e^{z}$, so if $p$ is non-constant, $f$ only omits $0$. Thus, $p$ must be constant, and thus, so is $f$.
\end{solution}

\begin{problem}[12.]
	Suppose $f$ is entire and never vanishes, and that none of the higher derivatives of $f$ ever vanish. Prove that if $f$ is also of finite order, then $f(z) = e^{az + b}$ for some constants $a$ and $b$.
\end{problem}

\begin{solution}
	Since $f$ is entire, never vanishes, and is of finite order, we have the Hadamard factorisation
	\[
		f(z) = e^{p(z)}
	\]
	for some polynomial $p$. By the chain rule,
	\[
		f'(z) = p'(z) e^{p(z)}
	\]
	but this has a zero unless $p'$ is a constant function. Thus, $p'$ is constant, and $p(z) = az + b$.
\end{solution}

\begin{problem}[13.]
	Show that the equation $e^{z} - z = 0$ has infinitely many solutions in $\mathbb{C}$.
\end{problem}

\begin{solution}
	Suppose by way of contradiction that $e^{z} - z$ has only finitely many zeroes in $\mathbb{C}$. Then, since $e^{z} - z \le 2 e^{\lvert z \rvert}$ has order at most $1$, by Hadamard's theorem we have
	\[
		e^{z} - z = e^{az + b} \prod_{n = 1}^{N} E_{1} \left ( \frac{z}{a_{n}} \right ) = e^{az + b} e^{\sum z / a_{n}} \prod_{n = 1}^{N} (1 - z / a_{n})
	\]
	where $a_{n}$ are the $N$ (non-zero) zeroes of $e^{z} - z$, counted with multiplicity. That is,
	\[
		e^{z} - z = e^{A z + B} p(z)
	\]
	where $p$ is a polynomial of degree $N$ and $A, B \in \mathbb{C}$ are some constants. We can take the derivative on both sides to get
	\[
		e^{z} - 1 = e^{Az + B} (A p(z) + p'(z)).
	\]
	Here, the left hand side is not identically zero, but has infinitely many zeroes, and the right hand side has either finitely many zeroes or is identically zero. Thus, we have a contradiction, and our supposition that $e^{z} - z$ has only finitely many zeroes must be false.
\end{solution}

\begin{problem}[7.]
	The \emph{Beta function} is defined for $\operatorname{Re} \alpha > 0$ and $\operatorname{Re} \beta > 0$ by
	\[
		B(\alpha, \beta) = \int_{0}^{1} (1 - t)^{\alpha - 1} t^{\beta - 1} \, dt.
	\]
	\begin{enumerate}
		\item Prove that
			\[
				B(\alpha, \beta) = \frac{\Gamma(a) \Gamma(b)}{\Gamma(a + b)}.
			\]
		\item Show that
			 \[
				B(\alpha, \beta) = \int_{0}^{\infty} \frac{u^{\alpha - 1}}{(1 + u)^{\alpha + \beta}} \, du.
			\]
	\end{enumerate}
\end{problem}
