\begin{problem}[1]
	Describe geometrically the sets of points $z$ in the complex plane defined by the following relations:
	\begin{enumerate}
		\item $\operatorname{Re}(az + b) > 0$ where $a, b \in \mathbb{C}$.
		\item $\lvert z \rvert = \operatorname{Re}(z) + 1$.
	\end{enumerate}
\end{problem}

\begin{solution}
	\begin{enumerate}
		\item Since $\operatorname{Re}$ is additive, we can write $\operatorname{Re}(az + b) = \operatorname{Re}(az) + \operatorname{Re} b$. Thus, we want $\operatorname{Re}(az) + \operatorname{Re} b > 0$. Suppose $a = a_{1} + a_{2} i$, $b = b_{1} + b_{2} i$, and $z = x + y i$. Then we're just looking for
			\[
				\begin{split}
					a_{1} x - a_{2} y + b_{1} > 0.
				\end{split}
			\]
			Geometrically, this is all points below the line
			 \[
				 y = \frac{a_{1} x  + b_{1}}{a_2}
			\]
			or, if $a_{2} = 0$, all points to the right of the line $x = -b_{1} / a_{1}$.
			Another way to describe the set is all $z$ such that the angle subtended between $z$ and the imaginary axis, after translation by $b$ is less than the angle subtended between $a$ and the real axis.
		\item Note that the equality is equivalent to squaring both sides as long as we make sure $\operatorname{Re} z + 1 \ge 0$. If we write $z = x + y i$, this is all $x + y i$ with $x > - 1$ and
			\[
				x^{2} + y^{2} = x^{2} + 2x + 1.
			\]
			Geometrically, this is all points $(x, y)$ with
			\[
				x = \frac{y^{2} - 1}{2}.
			\]
			Notice that we don't need to specify that $x > - 1$ since ${(y^{2} - 1)}/{2} \ge -1/2$ always. That is, this is all points on the parabola $x = \frac{y^{2} - 1}{2}$.
	\end{enumerate}
\end{solution}

\begin{problem}[5]
	A set $\Omega$ is said to be \emph{path-connected} if any two points in $\Omega$ can be joined by a (piecewise-smooth) curve entirely contained in $\Omega$. The purpose of this exercise is to prove that an \emph{open set} $\Omega$ is path-connected if and only if $\Omega$ is connected.
	\begin{enumerate}
		\item Suppose first that $\Omega$ is open and path-connected, and that it can be written as $\Omega = \Omega_{1} \cup \Omega_{2}$ where $\Omega_{1}$ and $\Omega_{2}$ are disjoint non-empty open sets. Choose two points $w_{1} \in \Omega_{1}$ and $w_{2} \in \Omega_{2}$, and let $\gamma$ denote a curve in $\Omega$ joining $w_{1}$ to $w_{2}$. Consider a parametrisation $z : [0, 1] \to \Omega$ of this curve with $z(0) = w_{1}$ and $z(1) = w_{2}$, and let
			\[
				t^{*} = \sup_{0 \le t \le 1} \{ t  : z(s) \in \Omega_{1} \text{ for all }0 \le s \le t \}
			\]
			Arrive at a contradiction by considering the point $z(t^{*})$.
		\item Conversely, suppose that $\Omega$ is open and connected. Fix a point $w \in \Omega$ and let $\Omega_{1} \subset \Omega$ denote the set of all points that can be joined to $w$ by a curve contained in $\Omega$. Also, let $\Omega_{2} \subset \Omega$ denote the set of all points that cannot be joined to $w$ by a curve in $\Omega$. Prove that both  $\Omega_{1}$ and $\Omega_{2}$ are open, disjoint, and their union is $\Omega$. Finally, since $\Omega_{1}$ is non-empty, conclude that $\Omega = \Omega_{1}$ as desired.
	\end{enumerate}
\end{problem}

\begin{solution}
	\begin{enumerate}
		\item We will show that $z(t^{*})$ cannot be in $\Omega_{1}$ and cannot be in $\Omega_{2}$ and thus, arrive at a contradiction. Let $T = \{ t : z(s) \in \Omega_{1} \text{ for all } 0 \le s \le t \}$.

			Suppose $z(t^{*})$ is in $\Omega_{1}$. Then, since $\Omega_{1}$ is open, there exists a neighbourhood  $N_{\varepsilon}(z(t^{*})) \subset \Omega_{1}$. Since $z$ is continuous, there exists some $\delta > 0$ such that $z^{\text{img}}((t^{*} - \delta, t^{*} + \delta))$ is contained in $N_{\varepsilon}(z(t^{*}))$, and thus, in $\Omega_{1}$. But then $t^{*} + \delta/2$ is in $T$ and is also greater than $t^{*}$. That is, $t^{*}$ is not an upper bound of $T$, contradicting our assumption that it is the supremum of $T$. Thus, $z(t^{*})$ cannot be in $\Omega_{1}$.

			Suppose $z(t^{*})$ is in $\Omega_{2}$. Then, again, since $\Omega_{2}$ is open, there exists a neighbourhood $N_{\varepsilon}(z(t^{*})) \subset \Omega_{2}$. Again, since $z$ is continuous, there exists some $\delta > 0$ such that $z^{\text{img}}((t^{*} - \delta, t^{*} + \delta))$ is contained in $N_{\varepsilon}(z(t^{*}))$, and thus, in $\Omega_{2}$. But then $t^{*} - \delta/2$ is a lesser upper bound of $T$ than $t^{*}$. This is because $t^{*}$ is an upper bound of $T$ and none of the reals between $t^{*} - \delta/2$ and $t^{*}$ are in $T$. But this contradicts that $t^{*}$ is the supremum of $T$. Thus, $z(t^{*})$ cannot be in $\Omega_{2}$ either.

			Since $z(t^{*}) \in \Omega$, but not in $\Omega_{1}$ or $\Omega_{2}$ which cover $\Omega$, we have a contradiction. Thus, our assumption that there exist disjoint open sets $\Omega_{1}$ and $\Omega_{2}$ with $\Omega = \Omega_{1} \cup \Omega_{2}$ must be false.
		\item 
			It's clear that $\Omega_{1} \sqcup \Omega_{2} = \Omega$ --- every point in $\Omega$ is either path-connected to $w$ or not. Now we show $\Omega_{1}$ and $\Omega_{2}$ are open.

			Suppose $w_{1} \in \Omega_{1}$. By the openness, of $\Omega$, we have some  $N_{r}(w_{1}) \subset \Omega$. Since this open neighbourhood is path connected, we can show that $N_{r}(w_{1}) \subset \Omega_{1}$ as well, by fusing the curve that connects $w$ to $w_{1}$ and the curve that connects $w_{1}$ and any other $w_{1}' \in N_{r}(w_{1})$ to form a piecewise smooth curve. Specifically, if $w$ and $w_{1}$ are connected by $\alpha$ parametrised by $z_{\alpha} : [0, 1] \to \Omega$ and $w_{1}$ and $w_{1}'$ are connected by $\beta$, parametrised by $z_{\beta} : [0, 1] \to N_{r}(w_{1})$, consider the curve $\gamma$ parametrised by
			\[
				z_{\gamma}(t) = \begin{cases}
					z_{\alpha}(2t) & t \le 1/2 \\
					z_{\beta}(2(t - 1/2)) & t > 1/2.
				\end{cases}
			\]
			$\gamma$ is piecewise smooth since $z_{\gamma}$ has at most one more non-smooth point than $z_{\alpha}$ and $z_{\beta}$ combined (at $t = 1/2$). Thus, $N_{r}(w_{1}) \subset \Omega_{1}$ as well, giving us that $\Omega_{1}$ is open.

			We can use a similar argument to show that $\Omega_{2}$ is open. For any $w_{2} \in \Omega_{2}$, consider $N_{r}(w_{2}) \subset \Omega$. Suppose by way of contradiction that there is some $w_{2}' \in N_{r}(w_{2}) \cap \Omega_{1}$ with a path $\alpha$ that connects $w$ to $w_{2}'$. Since $N_{r}(w_{2})$ is path-connected, $w_{2}'$ is connected to $w_{2}$ by a path $\beta$. Then, the path $\gamma$ parametrised by
			\[
				z_{\gamma}(t) = \begin{cases}
					z_{\alpha}(2t) & t \le 1/2 \\
					z_{\beta}(2(t - 1/2)) & t > 1/2.
				\end{cases}
			\]
			connects $w$ to $w_{2}$, contradicting our assumption that $w_{2} \in \Omega_{2}$. Thus, $N_{r}(w_{2}) \cap \Omega_{1}$ is empty, or equivalently, $N_{r}(w_{2}) \subset \Omega_{2}$. Thus, $\Omega_{2}$ is open.

			Note that $\Omega_{1}$ is non-empty since we $w$ is path-connected to itself. Specifically, given a neighbourhood $N_{r}(w) \subset \Omega$, the curve $\gamma$ parametrised by
			\[
				z_{\gamma}(t) = (w - r/3) + \frac{r}{3} e^{2 \pi t i}
			\]
			connects $w$ to itself.

			Since $\Omega$ is connected, any pair of disjoint open sets that covers $\Omega$ must have one empty set. We know $\Omega_{1}$ is non-empty, so $\Omega_{2}$ is empty. Thus, $\Omega_{1} = \Omega$. For any point $w \in \Omega$, every other point is path-connected to $w$. Thus, $\Omega$ is path-connected.
	\end{enumerate}
\end{solution}

\begin{problem}[7]
	The family of mappings introduced here plays an important role in complex analysis. These mappings, sometimes called \emph{Blaschke factors}, will reappear in various applications in later chapters.
	\begin{enumerate}
		\item Let $z, w$ be two complex numbers such that $\overline{z} w \neq 1$. Prove that
			\[
				\left \lvert \frac{w - z}{1 - \overline{w} z} \right \rvert < 1 \text{ if } \lvert z \rvert < 1 \text{ and } \lvert w \rvert < 1,
			\]
			and also that
			\[
					\left \lvert \frac{w - z}{1 - \overline{w} z} \right \rvert = 1 \text{ if } \lvert z \rvert = 1 \text{ or } \lvert w \rvert = 1,
			\]
		\item Prove that for a fixed $w$ in the unit disk $\mathbb{D}$, the mapping
			\[
				F : z \mapsto \frac{w - z}{1 - \overline{w} z}
			\]
			satisfies the following conditions:
			\begin{enumerate}[label = (\roman*)]
				\item $F$ maps the unit disk to itself and is holomorphic.
				\item $F$ interchanges $0$ and $w$.
				\item $\lvert F(z) \rvert = 1$ if $\lvert z \rvert = 1$.
				\item $F : \mathbb{D} \to \mathbb{D}$ is bijective.
			\end{enumerate}
	\end{enumerate}
\end{problem}

\begin{solution}
	\begin{enumerate}
		\item First we will show that it suffices to prove the case where $z$ is some real $r$. Suppose $a \in \mathbb{C}$ with $\lvert a \rvert = 1$ gives $z = ar \in \mathbb{R}$. That is, $a = e^{i \operatorname{arg} z}$. Also let $w' = w / a$. Then
			\[
				\begin{split}
					\left \lvert \frac{aw' - ar}{1 - \overline{aw'} (ar)} \right \rvert & = \lvert a \rvert \left \lvert \frac{w' - z}{1 - \lvert a \rvert^{2} \overline{w'} z} \right \rvert \\
													  & = \left \lvert \frac{w' - r}{1 - \overline{w'} r} \right \rvert
				\end{split}
			\]
			Thus, if we can show that the inequality holds in the case that $z$ is real (for $r$ and $w'$), we have the inequality for the general case of $z, w \in \mathbb{C}$. 

			Suppose we have $r \in \mathbb{R}$ and $w' \in \mathbb{C}$. Comparing $(r - w)(r - \overline{w})$ and $(1 - rw)(1 - r \overline{w})$ we see
			\[
				(r - w)(r - \overline{w}) = r^{2} - r w - r \overline{w} + \lvert w \rvert^{2}
			\]
			and
			\[
				(1 - r w)(1 - r \overline{w}) = 1 - rw - r\overline{w} + r^{2} \lvert w \rvert^{2}.
			\]
			The difference between these is
			\[
				\begin{split}
					(1 - rw)(1 - r\overline{w}) - (r - w)(r - \overline{w}) & = 1 + r^{2} \lvert w \rvert^{2} - r^{2} - \lvert w \rvert^{2} \\
												& = (1 - r^{2})(1 - \lvert w \rvert^{2})
				\end{split}	
			\]
			which is positive when $\lvert w \rvert < 1$ and $\lvert r \rvert < 1$ and zero when $\lvert w \rvert = 1$ or $\lvert r \rvert = 1$. That is
			\[
				\begin{split}
					\frac{(r - w)(r - \overline{w})}{(1 - rw)(1 - r\overline{w})} & = \frac{(r - w) \overline{(r - w)}}{(1 - r \overline{w}) \overline{(1 - r \overline{w})}} \\
												      & = \left \lvert \frac{w - r}{1 - \overline{w} r} \right \rvert \le 1
				\end{split}
			\]
			for $\lvert w \rvert, \lvert r \rvert \le 1$ with equality if and only if one of $\lvert w \rvert$ or $\lvert r \rvert$ is $1$. That is, we have the inequality in the case of real $z$, and thus, in the general case.
		\item
			\begin{enumerate}[label = (\roman*)]
				\item Since we have a fixed $w \in \mathbb{C}$ with $\lvert w \rvert < 1$, and any $z \in \mathbb{D}$ has $\lvert z \rvert \le 1$, by part (a), $\lvert F(z) \rvert \le 1$ for any $z \in \mathbb{D}$, and thus, $F(z) \in \mathbb{D}$.

					We can see that $F$ is holomorphic as it is the quotient of two holomorphic functions and $1 - \overline{w} z \neq 0$ since $\lvert \overline{w} z \rvert = \lvert w \rvert \lvert z \rvert < 1$.
				\item We just compute $F(0)$ and $F(w)$ ---
					\[
						\begin{split}
							F(0) & = \frac{w - 0}{1 - \overline{w} 0} \\
							     & = w/1 \\
							     & = w.
						\end{split}
					\]
					\[
						\begin{split}
							F(w) & = \frac{w - w}{1 - \overline{w} w} \\
							     & = \frac{0}{1 - \lvert w \rvert^{2}} \\
							     & = 0.
						\end{split}
					\]
				\item It follows directly from part (a) that if $\lvert z \rvert = 1$, $\lvert F(z) \rvert = 1$.
				\item We compute $F \circ F$.
					\[
						\begin{split}
							F \circ F(z) & = F \left ( \frac{w - z}{1 - \overline{w} z} \right ) \\
								     & = \frac{w - \frac{w - z}{1 - \overline{w} z}}{1 - \overline{w} \frac{w - z}{1 - \overline{w} z}} \\
								     & = \frac{w(1 - \overline{w} z) - w + z}{1 - \overline{w} z - \overline{w}(w - z)} \\
								     & = \frac{w - \lvert w \rvert^{2} z - w + z}{1 - \overline{w} z - \lvert w \rvert^{2} + \overline{w} z} \\
								     & = \frac{(1 - \lvert w \rvert^{2}) z}{(1 - \lvert w \rvert^{2})} \\
								     & = z.
						\end{split}
					\]
				Since $F$ is its own inverse, $F$ is bijective.
			\end{enumerate}
	\end{enumerate}
	
\end{solution}

\begin{problem}[9]
	Show that in polar coordinates, the Cauchy-Riemann equations take the form
	\[
		\frac{\partial u}{\partial r} = \frac{1}{r} \frac{\partial v}{\partial \theta} \text{ and } \frac{1}{r} \frac{\partial u}{\partial \theta} = - \frac{\partial v}{\partial r}.
	\]
	Use these equations to show that the logarithm function defined by
	\[
		\log z = \log r + i \theta \text{ where } z = r e^{i \theta} \text{ with } -\pi < \theta < \pi
	\]
	is holomorphic in the region $r > 0$ and $-\pi < \theta < \pi$.
\end{problem}

\begin{solution}
	If we compute all the partials, using the original Cauchy-Riemann equations we see that the polar form pops out. For the first equation ---
	\[
		\begin{split}
			\frac{\partial u}{\partial r} & = \frac{\partial u}{\partial x} \frac{\partial x}{\partial r} + \frac{\partial u}{\partial y} \frac{\partial y}{\partial r} \\
						      & = \frac{\partial u}{\partial x} \cos{\theta} + \frac{\partial u}{\partial y} \sin{\theta}
		\end{split}
	\]
	\[
		\begin{split}
			\frac{\partial v}{\partial \theta} & = \frac{\partial v}{\partial x} \frac{\partial x}{\partial \theta} + \frac{\partial v}{\partial y} \frac{\partial y}{\partial \theta} \\
							   & = - \frac{\partial v}{\partial x} r \sin{\theta} + \frac{\partial v}{\partial y} r \cos{\theta} \\
		\end{split}
	\]
	Substituting according to the Cauchy-Riemann equations we get
	\[
		\begin{split}
			\frac{\partial v}{\partial \theta} & = r \left ( \frac{\partial u}{\partial y} \sin{\theta} + \frac{\partial u}{\partial x} \cos{\theta} \right ) \\
							   & = r \frac{\partial u}{\partial r}
		\end{split}
	\]
	as desired.

	Similarly, for the second equation
	\[
		\begin{split}
			\frac{\partial u}{\partial \theta} & = \frac{\partial u}{\partial x} \frac{\partial x}{\partial \theta} + \frac{\partial u}{\partial \theta} \frac{\partial y}{\partial \theta} \\
						      & = - \frac{\partial u}{\partial x} r \sin{\theta} + \frac{\partial u}{\partial y} r \cos{\theta}
		\end{split}
	\]
	\[
		\begin{split}
			\frac{\partial v}{\partial \theta} & = \frac{\partial v}{\partial x} \frac{\partial x}{\partial r} + \frac{\partial v}{\partial y} \frac{\partial y}{\partial r} \\
							   & = \frac{\partial v}{\partial x} \cos{\theta} + \frac{\partial v}{\partial y} \sin{\theta} \\
		\end{split}
	\]
	Substituting according to the Cauchy-Riemann equations we get
	\[
		\begin{split}
			\frac{\partial v}{\partial \theta} & = \frac{1}{r} \left (- \frac{\partial u}{\partial y} r \cos{\theta} + \frac{\partial u}{\partial x} r \sin{\theta} \right ) \\
							   & = - \frac{1}{r} \frac{\partial u}{\partial \theta}
		\end{split}
	\]
	as desired.

	We can write the log function as $u + i v$ where $u(r e^{i \theta}) = \log r$ and $v(r e^{i \theta}) = \theta$. Thus, for $u_{r} = 1 / r$ which is continuous for $r > 0$ and $v_{\theta} = 1$ which is continuous for $-\pi < \theta < \pi$. Thus $u_{r} = v_{\theta} / r$. $u_{\theta} = v_{r} = 0$ always. That is, $u, v$ are continuously differentiable and satisfy the Cauchy-Riemann equations for $r > 0$ and $- \pi < \theta < \pi$. Thus, $\log$ must be holomorphic.
\end{solution}

\begin{problem}[11]
	Use Exercise 10 to prove that if $f$ is holomorphic in the open set $\Omega$, then the real and imaginary parts of $f$ are \emph{harmonic}; that is, their Laplacian is zero.
\end{problem}

\begin{solution}
	We already know that if $f$ is holomorphic, ${\partial f}/{\partial \overline{z}} = 0$. Let $f = u + i v$. Then,
	\[
		\begin{split}
			\Delta u + i \Delta v & = \Delta(u + iv) \\
					      & = \Delta f \\
					      & = 4 \frac{\partial }{\partial z} \left ( \frac{\partial f}{\partial \overline{z}} \right ) \\
					      & = 0.
		\end{split}
	\]
	We also know that if $f$ is holomorphic, $\partial f / \partial z = 2 \partial u/ \partial z$. Thus,
	\[
		\begin{split}
			\Delta u & = 4 \frac{\partial }{\partial \overline{z}} \left ( \frac{\partial u}{\partial z} \right ) \\
				 & = 2 \frac{\partial }{\partial \overline{z}} \left ( \frac{\partial f}{\partial z} \right ) \\
				 & = 2 \frac{\partial }{\partial z} \left ( \frac{\partial f}{\partial \overline{z}} \right ) \\
				 & = 0.
		\end{split}
	\]
	Thus, $0 + i \Delta v = 0$, and $\Delta v = 0$.
\end{solution}

\begin{problem}[13]
	Suppose that $f$ is holomorphic in a connected open set $\Omega$. Prove that in any one of the following cases:
	\begin{enumerate}
		\item $\operatorname{Re}(f)$ is constant;
		\item $\operatorname{Im}(f)$ is constant;
		\item $\lvert f \rvert$ is constant;
	\end{enumerate}
	one can conclude that $f$ is constant.
\end{problem}

\begin{solution}
	Write $f = u + iv$. Note that since $\Omega$ is open and connected, it is also path-connected with piecewise-smooth paths.
	\begin{enumerate}
		\item If $\operatorname{Re} f$ is constant, then $u$ is constant. Thus,
			 \[
				\frac{\partial u}{\partial x} = \frac{\partial u}{\partial y} = 0.
			\]
			By the Cauchy-Riemann equations, we must also have
			\[
				\frac{\partial v}{\partial y} = \frac{\partial u}{\partial x} = 0
			\]
			and
			\[
				\frac{\partial v}{\partial x} = - \frac{\partial u}{\partial y} = 0.
			\]
			Thus, $f' = {\partial f}/{\partial z} = 0$ everywhere, and by Corollary 3.4, $f$ is constant on  $\Omega$.
		\item If $\operatorname{Im} f$ is constant, then $v$ is constant. Thus,
			 \[
				\frac{\partial v}{\partial x} = \frac{\partial v}{\partial y} = 0.
			\]
			By the Cauchy-Riemann equations, we must also have
			\[
				\frac{\partial u}{\partial x} = \frac{\partial v}{\partial y} = 0
			\]
			and
			\[
				\frac{\partial u}{\partial y} = - \frac{\partial v}{\partial x} = 0.
			\]
			By the same argument as in part (b), $f$ must be constant.
		\item If $\lvert f \rvert$ is constant, so is $\lvert f \rvert^{2} = u^{2} + v^{2}$. Thus, 
			\[
				2 \left ( u \frac{\partial u}{\partial x} + v \frac{\partial v}{\partial x} \right ) = 0
			\]
			and
			\[
				2 \left ( u \frac{\partial u}{\partial y} + v \frac{\partial v}{\partial y} \right ) = 0.
			\]
			If $u$ or $v$ is constant, we are already done. Thus, we can assume $u, v \neq 0$. Thus,
			\[
				\begin{split}
					\frac{u}{v} \frac{\partial u}{\partial x} & = - \frac{\partial v}{\partial x} \\
										  & = \frac{\partial u}{\partial y}.
				\end{split}
			\]
			and
			\[
				\begin{split}
					\frac{u}{v} \frac{\partial u}{\partial y} & = - \frac{\partial v}{\partial y} \\
										  & = - \frac{\partial u}{\partial x}.
				\end{split}
			\]
			The linear system
			\begin{align*}
				\frac{u}{v} \frac{\partial u}{\partial x} - \frac{\partial u}{\partial y} & = 0 \\
				\frac{\partial u}{\partial x} + \frac{u}{v} \frac{\partial u}{\partial y} & = 0
			\end{align*}
			has the unique solution
			\[
				\frac{\partial u}{\partial x} = \frac{\partial u}{\partial y} = 0
			\]
			which we know, by part (a), will give $f$ constant.
	\end{enumerate}
\end{solution}
