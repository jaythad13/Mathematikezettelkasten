\begin{problem}[1]
	Describe geometrically the sets of points $z$ in the complex plane defined by the following relations:
	\begin{enumerate}
		\item $\operatorname{Re}(az + b) > 0$ where $a, b \in \mathbb{C}$.
		\item $\lvert z \rvert = \operatorname{Re}(z) + 1$.
	\end{enumerate}
\end{problem}

\begin{solution}
	\begin{enumerate}
		\item Since $\operatorname{Re}$ is additive, we can write $\operatorname{Re}(az + b) = \operatorname{Re}(az) + \operatorname{Re} b$. Thus, we want $\operatorname{Re}(az) + \operatorname{Re} b > 0$. Suppose $a = a_{1} + a_{2} i$, $b = b_{1} + b_{2} i$, and $z = x + y i$. Then we're just looking for
			\[
				\begin{split}
					a_{1} x - a_{2} y + b_{1} > 0.
				\end{split}
			\]
			Geometrically, this is all points below the line
			 \[
				 y = \frac{a_{1} x  + b_{1}}{a_2}.
			\]
			Another way to describe the set is all $z$ such that the angle subtended between $z$ and the imaginary axis, after translation by $b$ is less than the angle subtended between $a$ and the real axis.
		\item Note that the equality is equivalent to squaring both sides as long as we make sure $\operatorname{Re} z + 1 \ge 0$. If we write $z = x + y i$, this is all $x + y i$ with $x > - 1$ and
			\[
				x^{2} + y^{2} = x^{2} + 2x + 1.
			\]
			Geometrically, this is all points $(x, y)$ with
			\[
				x = \frac{y^{2} - 1}{2}.
			\]
			Notice that we don't need to specify that $x > - 1$ since ${(y^{2} - 1)}/{2} \ge -1/2$ always. That is, this is all points on the parabola $x = \frac{y^{2} - 1}{2}$.
	\end{enumerate}
\end{solution}

\begin{problem}[7]
	The family of mappings introduced here plays an important role in complex analysis. These mappings, sometimes called \emph{Blaschke factors}, will reappear in various applications in later chapters.
	\begin{enumerate}
		\item Let $z, w$ be two complex numbers such that $\overline{z} w \neq 1$. Prove that
			\[
				\left \lvert \frac{w - z}{1 - \overline{w} z} \right \rvert < 1 \text{ if } \lvert z \rvert < 1 \text{ and } \lvert w \rvert < 1,
			\]
			and also that
			\[
					\left \lvert \frac{w - z}{1 - \overline{w} z} \right \rvert = 1 \text{ if } \lvert z \rvert = 1 \text{ or } \lvert w \rvert < 1,
			\]
		\item Prove that for a fixed $w$ in the unit disk $\mathbb{D}$, the mapping
			\[
				F : z \mapsto \frac{w - z}{1 - \overline{w} z}
			\]
			satisfies the following conditions:
			\begin{enumerate}[label = (\roman*)]
				\item $F$ maps the unit disk to itself and is holomorphic.
				\item $F$ interchanges $0$ and $w$.
				\item $\lvert F(z) \rvert = 1$ if $\lvert z \rvert = 1$.
				\item $F : \mathbb{D} \to \mathbb{D}$ is bijective.
			\end{enumerate}
	\end{enumerate}
\end{problem}
