\begin{problem}[3.]
	Show, by contour integration, that if $a > 0$ and $\xi \in \mathbb{R}$, then
	\[
		\frac{1}{\pi} \int_{-\infty}^{\infty} \frac{a}{a^{2} + x^{2}} e^{-2 \pi i x \xi} \, dx = e^{-2 \pi a \lvert \xi \rvert}
	\]
	and check that
	\[
		\int_{-\infty}^{\infty} e^{-2 \pi a \lvert \xi \rvert} e^{2 \pi i \xi x} \, d\xi = \frac{1}{\pi} \frac{a}{a^{2} + x^{2}}
	\]
\end{problem}

\begin{solution} 
	Let $f : \mathbb{C} \setminus \{ \pm ai \} \to \mathbb{C}$ be the function by
	\[
		f : z \mapsto \frac{a e^{- 2 \pi i z \xi}}{a^{2} + x^{2}}
	\]
	$f$ is meromorphic with poles at $\pm ai$. We can calculate their residues as follows
	\[
		\begin{split}
			\operatorname{res}_{ai} f & = \lim_{z \to ai} (z - ai) \frac{ae^{-2 \pi i z \xi}}{a^{2} + z^{2}} \\
						  & = \lim_{z \to ai} \frac{a e^{-2 \pi i z \xi}}{z + ai} \\
						  & = \frac{a e^{- 2 \pi i (a i) \xi}}{2 a i} \\
						  & = \frac{-i e^{2 \pi a \xi}}{2}.
		\end{split}
	\]
	For $\xi < 0$, this is just $\operatorname{res}_{ai} f = - i e^{- 2 \pi a \lvert \xi \rvert} / 2$. By a similar calculation, $\operatorname{res}_{- ai} f = i e^{- 2 \pi a \lvert \xi \rvert} / 2$ for $\xi > 0$.

	For $\xi < 0$, the residue theorem says that the integral over the upper semicircle of radius $R$, $\gamma$ is
	\[
		\frac{1}{\pi} \int_{\gamma} f(z) \, dz = \frac{2 \pi i}{\pi} \operatorname{res}_{ai} f = e^{-2 \pi a \lvert \xi \rvert}.
	\]
	But as $R \to \infty$
	\[
		\int_{\gamma} f(z) \, dz = \int_{-\infty}^{\infty} f(x) \, dx + \int_{\gamma_{\text{arc}}} f(z) \, dz.
	\]
	where $\gamma_{\text{arc}}$ is the arc of the semicircle. Thus, if we show that
	\[
		\int_{\gamma_{\text{arc}}} f(z) \, dz \to 0
	\]
	then we have the desired result.

	Consider the parametrisation of $\gamma_{\text{arc}}$ by $z_{\gamma_{\text{arc}}} : \theta \mapsto R e^{i \theta}$ for $\theta \in [0, \pi]$. Then
	\[
		\begin{split}
			\left \lvert \int_{\gamma_{\text{arc}}} f(z) \, dz \right \rvert & = \left \lvert \int_{0}^{\pi} \frac{a e^{2 \pi R \lvert \xi \rvert (i \cos{\theta} - \sin{\theta})}}{a^{2} + R^{2} e^{2 i \theta}} i R e^{i \theta} \, d\theta \right \rvert \\
											 & \le \left \lvert \frac{R}{R^{2} - a^{2}} \right \rvert \left \lvert \int_{0}^{\pi} \lvert e^{2 \pi R \lvert \xi \rvert i \cos{\theta}} \rvert e^{- 2 \pi R \lvert \xi \rvert \sin{\theta}} \, d\theta  \right \rvert.
		\end{split}
	\]
	This goes to $0$ as $R \to \infty$ since the exponent in the integral is negative ($\sin{\theta}$ is positive for $\theta \in [0, \pi]$) , and the rational outside goes to $0$ as well. Thus,
	\[
		\int_{-\infty}^{\infty} \frac{a}{a^{2} + x^{2}} e^{-2 \pi i x \xi}  \, dx = e^{-2 \pi a \lvert \xi \rvert}
	\]
	for $\xi < 0$.

	To get the same result for $\xi > 0$ we can perform similar analysis for the integral over the lower semicircle. Here the residue theorem gives
	\[
		\int_{\infty}^{- \infty} f(x) \, dx + \int_{\text{arc}} f(z) \, dz = - e^{-2 \pi a \lvert \xi \rvert}.
	\]
	By a similar proof, we have that the integral over the arc vanishes as the radius of the semicircle goes to infinity, and thus,
	 \[
		\int_{\infty}^{-\infty} f(x) \, dx = - e^{-2 \pi a \lvert \xi \rvert}.
	\]
	That is,
	\[
		\int_{- \infty}^{\infty} f(x) \, dx = e^{-2 \pi a \lvert \xi \rvert}
	\]
	as desired.

	Now we compute the inversion formula ---
	\[
		\begin{split}
			\int_{-\infty}^{\infty} e^{-2 \pi a \lvert \xi \rvert} e^{2 \pi i \xi x} \, d \xi & = \int_{-\infty}^{0} e^{2 \pi (a + i x) \xi} \, d \xi + \int_{0}^{\infty} e^{- 2 \pi (a - i x) \xi} \, d \xi \\
													  & = \frac{1}{2 \pi (a + i x)} (e^{2 \pi (a + i x) \xi}) \Big |_{-\infty}^{0} - \frac{1}{2 \pi (a - i x)} (e^{- 2 \pi (a - i x) \xi}) \Big |_{0}^{\infty} \\
													  & = \frac{1}{2 \pi} \left ( \frac{1}{a + i x} + \frac{1}{a - i x} \right ) \\
													  & = \frac{1}{\pi} \frac{a}{a^{2} + x^{2}}.
		\end{split}
	\]
	as desired.
\end{solution}

\begin{problem}[4.]
	Suppose $Q$ is a polynomial of degree $\ge 2$ with distinct roots, none lying on the real axis. Calculate
	\[
		\int_{-\infty}^{\infty} \frac{e^{-2 \pi i x \xi}}{Q(x)} \, dx, \quad \xi \in \mathbb{R}
	\]
	in terms of the roots of $Q$. What happens when several roots coincide?
\end{problem}

\begin{solution}
	For $Q(z) = a \prod_{i = 1}^{n} (z - e_{i})$
	\[
		\boxed{
			\int_{-\infty}^{\infty} \frac{e^{-2 \pi i x \xi}}{Q(x)} \, dx = \begin{cases}
				\displaystyle 2 \pi i \sum_{\operatorname{Im} e_{i} > 0} \frac{e^{2 \pi i \xi e_{i}}}{a \prod_{j \neq i} (e_{i} - e_{j})} & \xi < 0 \\
				\displaystyle - 2 \pi i \sum_{\operatorname{Im} e_{i} < 0} \frac{e^{2 \pi i \xi e_{i}}}{a \prod_{j \neq i} (e_{i} - e_{j})} & \xi > 0.
			\end{cases}
		}
	\]

	The function $f : \mathbb{C} \setminus \{ e_{1}, \dots, e_{n} \} \to \mathbb{C}$ by $f(z) = e^{-2 \pi i z \xi} / Q(z)$ is meromorphic with poles at the $e_{1}, \dots, e_{n}$. Using the same trick as before, we can evaluate the integral over upper and lower half semicircles depending on the sign of $\xi$. Thus, we calculate the residues at each root.
	\[
		\begin{split}
			\operatorname{res}_{e_{i}} f & = \lim_{z \to e_{i}} (z - e_{i}) \frac{e^{-2 \pi i z \xi}}{a \prod_{j = 1}^{n} (z - e_{j})} \\
						     & = \frac{e^{2 \pi i \xi e_{i}}}{a \prod_{j \neq i} (e_{i} - e_{j})}.
		\end{split}
	\]

	Thus, the integral of $f$ over the upper half semicircle $\gamma$ is
	\[
		\int_{\gamma} f(z) \, dz = 2 \pi i \sum_{\operatorname{Im} e_{i} > 0} \frac{e^{2 \pi i \xi e_{i}}}{a \prod_{j \neq i} (e_{i} - e_{j})}.
	\]
	Again, this means that we can write
	\[
		2 \pi i \sum_{\operatorname{Im} e_{i} > 0} \frac{e^{2 \pi i \xi e_{i}}}{a \prod_{j \neq i} (e_{i} - e_{j})} = \int_{\gamma_{\text{arc}}} f(z) \, dz + \int_{-\infty}^{\infty} f(x) \, dx
	\]
	where $\gamma_{\text{arc}}$ is the semicircular arc parametrised by $z : \theta \mapsto e^{i \theta}$. If $\xi < 0$ we can show that the integral over $\gamma_{\text{arc}}$ goes to $0$.
	\begin{align*}
		\left \lvert \int_{\gamma_{\text{arc}}} f(z) \, dz \right \rvert & = \left \lvert \int_{0}^{\pi} \frac{e^{-2 \pi i \xi R (\cos{\theta} + i \sin{\theta})}}{Q(R e^{i \theta})} \, dz \right \rvert \\
										 & \le \frac{1}{\lvert Q(R e^{i \theta}) \rvert} \left \lvert \int_{0}^{\pi} e^{2 \pi \xi R \sin{\theta}} \lvert e^{-2 \pi i \xi R \cos{\theta}} \rvert \, d\theta \right \rvert \\
										 & \le \frac{1}{Q(\lvert R e^{i \theta}) \rvert} \left \lvert \int_{0}^{\pi} e^{2 \pi \xi R \sin{\theta}} \, d\theta 
 \right \rvert.
	\end{align*}
	For $\xi < 0$, the integral goes to zero as $R \to \infty$ by Jordan's inequality, so we're left with
	\[
		\int_{-\infty}^{\infty} \frac{e^{-2 \pi i x \xi}}{Q(x)} \, dx = 2 \pi i \sum_{\operatorname{Im} e_{i} > 0} \frac{e^{2 \pi i \xi e_{i}}}{a \prod_{j \neq i} (e_{i} - e_{j})}
	\]

	Similar analysis using the lower semicircle in the lower semicircle gives us the corresponding result for $\xi > 0$. The sign is reversed because the integral goes counter-clockwise from $R$ to $-R$.
\end{solution}

\begin{problem}[7. (a)]
	The Poisson summation formula applied to specific examples often provides interesting identities.

	Let $\tau$ be fixed with $\operatorname{Im} \tau > 0$. Apply the Poisson summation formula to
	\[
		f(z) = (\tau + z)^{-k}
	\]
	where $k$ is an integer $\ge 2$ to obtain
	\[
		\sum_{n = -\infty}^{\infty} \frac{1}{(\tau + n)^{k}} = \frac{(-2 \pi i)^{k}}{(k - 1)!} \sum_{m = 1}^{\infty} m^{k - 1} e^{2 \pi i m \tau}.
	\]
\end{problem}

\begin{solution}
	The $m$th Fourier coefficient is given by the integral
	\[
		\int_{-\infty}^{\infty} \frac{e^{-2 \pi i m x}}{(\tau + x)^{k}} \, dx.
	\]
	If we write
	\[
		g_{m}(z) = \frac{e^{- 2 \pi i m z}}{(\tau + z)^{k}}
	\]
	we can evaluate the Fourier coefficients in terms of the residues of $g$. $g_{m}$ has only one pole --- a pole of order $k$ at $- \tau$ in the lower half plane. At that pole, it has residue
	\[
		\begin{split}
			\operatorname{res}_{- \tau} g_{m} & = \frac{1}{(k - 1)!} \lim_{z \to - \tau} \frac{d^{k - 1}}{dz^{k - 1}} \left ( \frac{e^{-2 \pi i m z}}{(\tau + z)^{k}} (\tau + z)^{k} \right ) \\
							  & = \frac{1}{(k - 1)!} \lim_{z \to - \tau} (-2 \pi i m)^{k - 1} e^{-2 \pi i m z} \\
							  & = \frac{(- 2 \pi i)^{k - 1}}{(k - 1)!} e^{2 \pi i m \tau}.
		\end{split}
	\]

	For $m > 0$, in order to evaluate the Fourier coefficient, we consider the integral of $g_{m}$ over $\gamma$, the lower half semicircle of radius $R$ and take limits as $R \to \infty$. We know by the residue formula that for large enough $R$ (all $R > \lvert \tau \rvert$)
	\[
		\int_{\gamma} g_{m}(z) \, dz = 2 \pi i \operatorname{res}_{- \tau} g_{m} = - \frac{(- 2 \pi i)^{k - 1}}{(k - 1)!} e^{2 \pi i m \tau}.
	\]
	
	We can also write the integral as the sum of the Fourier coefficient and the integral of $g_{m}$ over the semicircular arc $\gamma_{\text{arc}}$.
	\[
		\begin{split}
			- \frac{(- 2 \pi i )^{k}}{(k - 1)!} e^{2 \pi i m \tau} & =  \int_{\gamma} g_{m}(z) \, dz \\
									       & = - \int_{\gamma_{\text{arc}}} g_{m}(z) \, dz - \int_{-\infty}^{\infty} g_{m}(x) \, dx
		\end{split}
	\]

	We show that the integral over the arc vanishes as we did in previous problems. With parametrisation of the reverse arc by $z : \theta \mapsto R e^{- i \theta}$ for $\theta \in [0, \pi]$, we get
	\[
		\begin{split}
			\left \lvert \int_{\gamma_{\text{arc}}} g_{m}(z) \, dz \right \rvert & = \left \lvert \int_{0}^{\pi} \frac{e^{- 2 \pi i m R(\cos{\theta} - i \sin{\theta})}}{(\tau + R e^{i \theta})^{k}} \, dz \right \rvert \\
											     & \le \frac{1}{(R - \lvert \tau \rvert)^{k}} \left \lvert \int_{0}^{\pi} e^{-2 \pi i m R \cos{\theta}} e^{- 2 \pi m R \sin{\theta}} \, d\theta  \right \rvert \\
											     & \le \frac{1}{(R - \lvert \tau \rvert)^{k}} \left \lvert \int_{0}^{\pi} e^{- 2 \pi m R \sin{\theta}} \, d \theta \right \rvert.
		\end{split}
	\]
	As $R \to \infty$, this integral goes to $0$.

	This leaves us with
	\[
		\int_{-\infty}^{\infty} g_{m}(x) \, dx = \frac{(-2 \pi i)^{k}}{(k - 1)!} e^{2 \pi i m \tau}
	\]
	for $m > 0$, as desired.

	For $m \le 0$, we consider the integral over the upper semicircle as its radius $R \to \infty$. This encloses no residue, so the total integral is zero. Again, by a similar analysis as in the previous part of the problem, the integral over the arc vanishes (the $m  = 0$ case is slightly different, but again is a simple application of the triangle inequality). Thus, we're left with
	\[
		\int_{-\infty}^{\infty} g_{m}(x) \, dx = 0 - 0 = 0
	\]
	for $m \le 0$.

	Thus, all the Fourier coefficients are the necessary values for the Poisson summation formula to give
	\[
		\sum_{n = -\infty}^{\infty} \frac{1}{(\tau + n)^{k}} = \frac{(-2 \pi i)^{k}}{(k - 1)!} \sum_{m = 1}^{\infty} m^{k - 1} e^{2 \pi i m \tau}.
	\]
\end{solution}

\begin{problem}[8.]
	Suppose $\hat{f}$ has compact support contained in $[-M, M]$ and let $f(z) = \sum_{n = 0}^{\infty} a_{n} z^{n}$. Show that
	\[
		a_{n} = \frac{(2 \pi i)^{n}}{n!} \int_{-M}^{M} \hat{f}(\xi) \xi^{n} \, d\xi
	\]
	and as a result
	\[
		\limsup_{n \to \infty} (n! \lvert a_{n} \rvert)^{1/n} \le 2 \pi M.
	\]
\end{problem}

\begin{solution}
	We know that $a_{n}$ is just $f^{(n)}(0) / n!$. By the Fourier inversion formula, this is
	\[
		\begin{split}
			a_{n} & = \frac{f^{(n)}(0)}{n!} \\
			      & = \frac{1}{n!} \frac{d^{n}}{dz^{n}} \left ( \int_{-M}^{M} \hat{f}(\xi) e^{2 \pi i z \xi} \, d\xi \right ) \Big |_{z = 0} \\
			      & = \frac{1}{n!} \int_{-M}^{M} \hat{f}(\xi) \left ( \frac{\partial^{n}(e^{2 \pi i z \xi})}{\partial z^{n}} \right ) \Big |_{z = 0} \, d\xi \\
			      & = \frac{1}{n!} \int_{-M}^{M} \hat{f}(\xi) (2 \pi i \xi)^{n} e^{2 \pi i \xi 0} \, d\xi \\
			      & = \frac{(2 \pi i)^{n}}{n!} \int_{-M}^{M} \hat{f}(\xi) \xi^{n} \, d\xi.
		\end{split}
	\]
	Here, moving the derivative under the integral sign is justified by the function $e^{2 \pi i z \xi}$ having continuous partials with respect to $z$.
	
	We can bound $a_{n}$ by the triangle inequality ---
	\[
		\begin{split}
			(n! \lvert a_{n} \rvert)^{1/n} & = \left \lvert (2 \pi i)^{n} \int_{-M}^{M} \hat{f}(\xi) \xi^{n} \, d\xi  \right \rvert^{1/n} \\
						       & \le 2 \pi M \lvert \hat{f}(\xi) \rvert^{1/n}
		\end{split}
	\]
	For large enough $n$, $\lvert \hat{f}(\xi) \rvert^{1/n} \le 1$, and thus,
	\[
		\limsup_{n \to \infty} (n! \lvert a_{n} \rvert)^{1/n} \le 2 \pi M.
	\]
\end{solution}
