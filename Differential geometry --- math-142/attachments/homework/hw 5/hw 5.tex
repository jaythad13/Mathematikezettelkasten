\begin{problem}[Problem 1.]
	 For the curve $\alpha : \mathbb{R} \to \mathbb{R}^{3}$  defined by $\alpha(t) = (2t, t^{2}, t^{3}/3)$ ,
	\begin{enumerate}
		\item Compute the Frenet-Serret apparatus.
		\item Sketch the curve for $-4 \le t \le 4$, showing $\mathbf{T}$, $\mathbf{N}$, and $\mathbf{B}$ at $t = 2$.
		\item Find the limiting values of $\mathbf{T}$, $\mathbf{N}$, and $\mathbf{B}$ as $t \to - \infty$ and $t \to \infty$.
	\end{enumerate} 
\end{problem}

\begin{solution}
	For this exercise let $\mathbf{T}, \mathbf{N}, \mathbf{B}, \alpha', \alpha'', \alpha'''$ denote just the vector parts of the tangent vectors that are usually denoted by them.
	\begin{enumerate}
		\item
			\[
				\boxed{
					\begin{gathered}
						\mathbf{T} = \left ( \frac{2}{t^{2} + 2}, \frac{2t}{t^{2} + 2}, 1 - \frac{2}{t^{2} + 2} \right ). \\
						\mathbf{N} = \frac{(-2t, 2 - t^{2}, 2t)}{t^{2} + 2}. \\
						\mathbf{B} = \frac{(t^{2}, -2t, 2)}{t^{2} + 2}. \\
						v = t^{2} + 2. \\
						\kappa = \frac{2}{(t^{2} + 2)^{2}}. \\
						\tau = \frac{2}{(t^{2} + 2)^{2}}.
					\end{gathered}
				}
			\] \\
			Given $\alpha$, we can compute $\alpha'$, $\alpha''$, and $\alpha'''$ as follows.
			\[
				\begin{gathered}
					\alpha' = (2, 2t, t^{2}) \\
					\alpha'' = (0, 2, 2t) \\
					\alpha''' = (0, 0, 2)
				\end{gathered}
			\]
			Thus, we have the speed $v$ given by
			\[
				\begin{split}
					v & = \lVert \alpha' \rVert \\
					  & = \lVert (2, 2t, t^{2}) \rVert \\
					  & = \sqrt{4 + 4t^{2} + t^{4}} \\
					  & = t^{2} + 2.
				\end{split}
			\]
			This also gives us
			\[
				\begin{split}
					\mathbf{T} & = \frac{\alpha'}{v} \\
						   & = \frac{(2, 2t, t^{2})}{t^{2} + 2} \\
						   & = \left ( \frac{2}{t^{2} + 2}, \frac{2t}{t^{2} + 2}, 1 - \frac{2}{t^{2} + 2} \right ).
				\end{split}
			\] \\
			Now, we can calculate $\mathbf{N}$ by looking at $\mathbf{T}'$ and its norm ---
			\[
				\begin{split}
					\mathbf{T}' & = \frac{d}{dt} \left ( \frac{2}{t^{2} + 2}, \frac{2t}{t^{2} + 2}, 1 - \frac{2}{t^{2} + 2} \right ) \\
						    & = \left ( - \frac{4t}{(t^{2} + 2)^{2}}, \frac{2}{t^{2} + 2} - \frac{4t^{2}}{(t^{2} + 2)^{2}}, \frac{4t}{(t^{2} + 2)^{2}} \right ) \\
						    & = \left ( - \frac{4t}{(t^{2} + 2)^{2}}, \frac{4 - 2t^{2}}{(t^{2} + 2)^{2}}, \frac{4t}{(t^{2} + 2)^{2}} \right ) \\
						    & = \frac{(-4t, 4 - 2t^{2}, 4t)}{(t^{2} + 2)^{2}}.
				\end{split}
			\]
			Thus, we have $\lVert \mathbf{T}' \rVert$ given by
			\[
				\begin{split}
					\lVert \mathbf{T}' \rVert & = \sqrt{\frac{4t^{4} + 16t^{2} + 16}{(t^{2} + 2)^{4}}} \\
								 & = \sqrt{\frac{4(t^{2} + 2)^{2}}{(t^{2} + 2)^{4}}} \\
								 & = \frac{2}{t^{2} + 2}
				\end{split}
			\]
			Thus, we have
			\[
				\begin{split}
					\mathbf{N} & = \frac{\mathbf{T}'}{\lVert \mathbf{T}' \rVert} \\
						   & = \frac{(-2t, 2 - t^{2}, 2t)}{t^{2} + 2}.
				\end{split}
			\] \\
			From this, we can calculate $\mathbf{B} = \mathbf{T} \times \mathbf{N}$ as follows 
			\[
				\begin{split}
					\mathbf{B} & = \mathbf{T} \times \mathbf{N} \\
						   & = \frac{(2, 2t, t^{2})}{t^{2} + 2} \times \frac{(-2t, 2 - t^{2}, 2t)}{t^{2} + 2} \\
						   & = \frac{1}{(t^{2} + 2)^{2}}
						   \begin{vmatrix}
							   \mathbf{i} & \mathbf{j} & \mathbf{k} \\
							   2 & 2t & t^{2} \\
							   -2t & 2 - t^{2} & 2t
						   \end{vmatrix} \\
						   & = \frac{(t^{4} + 2t^{2}, -2t^{3} - 4t, 2t^{2} + 4)}{(t^{2} + 2)^{2}} \\
						   & = \frac{(t^{2}, -2t, 2)}{t^{2} + 2}.
				\end{split}
			\]
			Finally, we can calculate $\kappa$ as
			\[
				\begin{split}
					\kappa & = \frac{\lVert (2, 2t, t^{3}) \times (0, 2, 2t) \rVert}{(t^{2} + 2)^{3}} \\
					       & = \frac{1}{(t^{2} + 2)^{2}}
					       \begin{Vmatrix}
						       \mathbf{i} & \mathbf{j} & \mathbf{k} \\
						       2 & 2t & t^{2} \\
						       0 & 2 & 2t
					       \end{Vmatrix} \\
					       & = \frac{\lVert (2t^{2}, - 4t, 4) \rVert}{(t^{2} + 2)^{3}} \\
					       & = \frac{\sqrt{4t^{4} + 16 t^{2} + 16}}{(t^{2} + 2)^{2}} \\
					       & = \frac{2 (t^{2} + 2)}{(t^{2} + 2)^{3}} \\
					       & = \frac{2}{(t^{2} + 2)^{2}}.
				\end{split}
			\] \\
			Note that this gives us the result $\lVert \alpha' \times \alpha'' \rVert = 2(t^{2} + 2)$ for $\tau$
			\[
				\begin{split}
					\tau & = \frac{\alpha' \times \alpha'' \cdot \alpha'''}{\lVert \alpha' \times \alpha'' \rVert^{2}} \\
					     & = \frac{(2t^{2}, -4t, 4) \cdot (0, 0, 2)}{4(t^{2} + 2)^{2}} \\
					     & = \frac{2}{(t^{2} + 2)^{2}}.
				\end{split}
			\]
		\item
			\begin{figure}[H]
				\centering
				\includegraphics[width = 0.75 \textwidth]{P 1 b curve with Frenet frame}
				\caption{$\alpha$ for $t \in [-4, 4]$ with the Frenet frame at $t = 2$}
			\end{figure}
		\item As $t \to \infty$ we have
			\[
				\boxed{
					\begin{gathered}
						\mathbf{T} \to (0, 0, 1) \\
						\mathbf{N} \to (0, -1, 0) \\
						\mathbf{B} \to (1, 0, 0).
					\end{gathered}
				}
			\]
			and as $t \to \infty$
			\[
				\boxed{
					\begin{gathered}
						\mathbf{T} \to (0, 0, 1) \\
						\mathbf{N} \to (0, -1, 0) \\
						\mathbf{B} \to (1, 0, 0).
					\end{gathered}
				}
			\]
	\end{enumerate}
\end{solution}

\begin{problem}[Problem 2.]
	We say that a regular curve $\alpha : I \to \mathbb{R}^{3}$ is a \textit{cylindrical helix} if the unit tangent vector $\mathbf{T}$ has a constant angle for some fixed unit tangent vector $[\mathbf{u}, \mathbf{p}] \in \mathrm{T}_{\mathbf{p}}\mathbb{R}^{3}$; that is, $\mathbf{T} \cdot [\mathbf{u}, \mathbf{p}] = \cos{\vartheta}$ whenever $\mathbf{p} = \alpha(t)$.
	\begin{enumerate}
		\item If $\alpha : I \to \mathbb{R}^{3}$ is a cylindrical helix, prove that the unit vector $[\mathbf{u}, \mathbf{p}]$ is
			\[
				\frac{\tau}{\sqrt{\kappa^{2} + \tau^{2}}} \mathbf{T} + \frac{\kappa}{\sqrt{\kappa^{2} + \tau^{2}}} \mathbf{B}.
			\]
		\item Check (a) for the cylindrical helix $\alpha : \mathbb{R} \to \mathbb{R}^{3}$ defined by $\alpha(t) = (a \cos{t}, a \sin{t}, bt)$ with $a, b \in \mathbb{R} \setminus \{ 0 \}$.	
	\end{enumerate}
\end{problem}

\begin{solution}
	\begin{enumerate}
		\item	Let $[\mathbf{u}, \mathbf{p}]$ be the unit constant vector with fixed angle to the tangent vector. Then note that we can show that $[\mathbf{u}, \mathbf{p}]$ has only constant components in the directions of $\mathbf{T}$ and $\mathbf{B}$. \\\\
			Specifically,
			\[
				\begin{split}
					(\mathbf{T} \cdot [\mathbf{u}, \mathbf{p}])' & = \mathbf{T}' \cdot [\mathbf{u}, \mathbf{p}] + \mathbf{T} \cdot [\mathbf{u}, \mathbf{p}]' \\
										     & = \kappa v \, \mathbf{N} \cdot [\mathbf{u}, \mathbf{p}]
				\end{split}
			\]
			But we already know that $\mathbf{T} \cdot [\mathbf{u}, \mathbf{p}]$ is constant and thus, that $(\mathbf{T} \cdot [\mathbf{u}, \mathbf{p}])' = 0$. Thus, $\kappa v \, \mathbf{N} \cdot [\mathbf{u}, \mathbf{p}] = 0$ always, and since $\alpha$ is strongly regular $\kappa v$ is never $0$. Thus, $\mathbf{N} \cdot [\mathbf{u}, \mathbf{p}] = 0$ and $[\mathbf{u}, \mathbf{p}]$ has no component along $\mathbf{N}$. \\\\
			Since we know that $\mathbf{T} \cdot [\mathbf{u}, \mathbf{p}]$ and $\lVert [\mathbf{u}, \mathbf{p}] \rVert = 1$, we can write $\lVert [\mathbf{u}, \mathbf{p}] \rVert$ in the orthonormal basis $\mathbf{T}, \mathbf{N}, \mathbf{B}$ to get
			\[
				(\mathbf{T} \cdot [\mathbf{u}, \mathbf{p}])^{2} + (\mathbf{N} \cdot [\mathbf{u}, \mathbf{p}])^{2} + (\mathbf{B} \cdot [\mathbf{u}, \mathbf{p}])^{2} = 1
			\]
			which reduces to
			\[
				(\mathbf{B} \cdot [\mathbf{u}, \mathbf{p}])^{2} = 1 - \cos^{2}{\vartheta}.
			\]
			Since $\cos{\vartheta}$ is constant, we get that $\mathbf{B} \cdot [\mathbf{u}, \mathbf{p}] = \sin{\vartheta}$ is constant. Thus,
			\[
				[\mathbf{u}, \mathbf{p}] = \cos{\vartheta} \, \mathbf{T} + \sin{\vartheta} \, \mathbf{B}.
			\] \\
			Once we have this result we can differentiate on both sides with respect to $t$ to get
			\[
				\begin{split}
					0 & = [\mathbf{u}, \mathbf{p}]' \\
					  & = \cos{\vartheta} \, \mathbf{T}' + \sin{\vartheta} \, \mathbf{B}' \\
					  & = \kappa v \cos{\vartheta} \, \mathbf{N} - \tau v \sin{\vartheta} \, \mathbf{N}
				\end{split}
			\]
			Thus,
			\[
				\kappa v \cos{\vartheta} \, \mathbf{N} = \tau v \sin{\vartheta} \, \mathbf{N}
			\]
			and since none of $v, \mathbf{N}$ are zero, we have
			\[
				\kappa \cos{\vartheta} = \tau \sin{\vartheta}.
			\] \\
			Squaring both sides, we get
			\[
				\kappa^{2} \cos^{2}{\vartheta} = \tau^{2} (1 - \cos^{2}{\vartheta})
			\]
			which we can rewrite as
			\[
				\cos^{2}{\vartheta} = \frac{\tau^{2}}{\kappa^{2} + \tau^{2}}.
			\]
			Note that this also gives us
			\[
				\begin{split}
					\sin^{2}{\vartheta} & = 1 - \cos^{2}{\vartheta} \\
							    & = 1 - \frac{\tau^{2}}{\kappa^{2} + \tau^{2}} \\
							    & = \frac{\kappa^{2}}{\kappa^{2} + \tau^{2}} 
				\end{split}
			\] \\
			Taking square roots we get
			\[
				\begin{gathered}
					\cos{\vartheta} = \frac{\tau}{\sqrt{\kappa^{2} + \tau^{2}}} \\
					\sin{\vartheta} = \frac{\kappa}{\sqrt{\kappa^{2} + \tau^{2}}}
				\end{gathered}
			\]
			and thus
			\[
				[\mathbf{u}, \mathbf{p}] = \frac{\tau}{\sqrt{\kappa^{2} + \tau^{2}}} \mathbf{T} + \frac{\kappa}{\sqrt{\kappa^{2} + \tau^{2}}} \mathbf{B}
			\]
			as desired.
		\item For $\alpha(t) = (a \cos{t}, a \sin{t}, bt)$ we have 
			\[
				\begin{gathered}
					\alpha' = (- a \sin{t}, a \cos{t}, b) \\
					\alpha'' = (- a \cos{t}, - a \sin{t}, 0) \\
					\alpha''' = (a \sin{t}, - a \cos{t}, 0)
				\end{gathered}
			\]
			and thus
			\[
				\mathbf{T} = \frac{(- a \sin{t}, a \cos{t}, b)}{\sqrt{a^{2} + b^{2}}}.
			\]
			We also have
			\[
				\begin{split}
					\kappa & = \frac{\lVert (- a \sin{t}, a \cos{t}, b) \times (- a \cos{t}, - a \sin{t}, 0) \rVert}{(\sqrt{a^{2} + b^{2}})^{3}} \\
					       & = \frac{\lVert (ab \sin{t}, -ab \cos{t}, a^{2}) \rVert}{(\sqrt{a^{2} + b^{2}})^{3}} \\
					       & = \frac{a \sqrt{b^{2} + a^{2}}}{(\sqrt{a^{2} + b^{2}})^{3}} \\
					       & = \frac{a}{a^{2} + b^{2}}.
				\end{split}
			\]
			Similarly
			\[
				\begin{split}
					\tau & = \frac{(- a \sin{t}, a \cos{t}, bt) \times (-a \cos{t}, - a \sin{t}, 0) \cdot (a \sin{t}, -a \cos{t}, 0)}{a^{2}(a^{2} + b^{2})} \\
					     & = \frac{(ab \sin{t}, - ab \cos{t}, a^{2}) \cdot (a \sin{t}, -a \cos{t}, 0)}{a^{2}(a^{2} + b^{2})} \\
					     & = \frac{a^{2}b(\sin^{2}{t} + \cos^{2}{t})}{a^{2}(a^{2} + b^{2})} \\
					     & = \frac{b}{a^{2} + b^{2}}.
				\end{split}
			\]
			We can also calculate $\mathbf{B}$ as
			\[
				\begin{split}
					\mathbf{B} & = \frac{(- a \sin{t}, a \cos{t}, b) \times (- a \cos{t}, - a \sin{t}, 0)}{\lVert (- a \sin{t}, a \cos{t}, b) \times (- a \cos{t}, - a \sin{t}, 0) \rVert} \\
						   & = \frac{(ab \sin{t}, - ab \cos{t}, a^{2})}{a \sqrt{a^{2} + b^{2}}}
				\end{split}
			\]
			Now we can check whether $[\mathbf{u}, \alpha(t)]$ is constant. We compute
			 \[
				\begin{split}
					\mathbf{u} & = \frac{\tau}{\sqrt{\tau^{2} + \kappa^{2}}} \mathbf{T} + \frac{\kappa}{\tau^{2} + \kappa^{2}} \mathbf{B} \\
						   & = \frac{b}{\sqrt{a^{2} + b^{2}}} \frac{(- a \sin{t}, a \cos{t}, b)}{\sqrt{a^{2} + b^{2}}} + \frac{a}{\sqrt{a^{2} + b^{2}}} \frac{(ab \sin{t}, - ab \cos{t}, a^{2})}{a \sqrt{a^{2} + b^{2}}} \\
						   & = (0, 0, 1).
				\end{split}
			\]
			and thus, $\mathbf{u}$ is constant, and it satisfies the condition that
			\[
				\mathbf{T} \cdot [\mathbf{u}, \alpha(t)] = \frac{b}{\sqrt{a^{2} + b^{2}}}
			\]
			is constant, as desired.
	\end{enumerate}
\end{solution}

\begin{problem}[Problem 3.]
	Let $\alpha : I \to \mathbb{R}^{3}$ be a cylindrical helix with unit vector $[\mathbf{u}, \alpha(t)]$. For each fixed $t_{0} \in I$, the curve $\sigma : I \to \mathbb{R}^{3}$ defined by
	\[
		\sigma(t) = \alpha(t) - ((\alpha(t) - \alpha(t_{0})) \cdot \mathbf{u}) \mathbf{u}
	\]
	is called a \textit{cross-sectional curve} of the cylinder on which $\alpha$ lies. Prove:
	\begin{enumerate}
		\item $\sigma$ lies on the plane $\mathcal{P} = \{ (x, y, z) \in \mathbb{R}^{3} | u_{1}(x - x_{0}) + u_{2}(y - y_{0}) + u_{3}(z - z_{0}) = 0 \}$ where $\mathbf{u} = (u_{1}, u_{2}, u_{3})$ and $\alpha(t_{0}) = (x_{0}, y_{0}, z_{0})$. 
		\item The curvature of $\sigma$ is $\kappa/\sin^{2}{\vartheta}$ where $\kappa$ is the curvature of $\alpha$.
	\end{enumerate}
\end{problem}

\begin{solution}
	\begin{enumerate}
		\item It suffices to show that $\sigma(t) - \alpha(t_{0})$  is always orthogonal to $\mathbf{u}$ . We can do this by computing
			\[
				\begin{split}
					(\sigma(t) - \alpha(t_{0})) \cdot \mathbf{u} & = (\alpha(t) - \alpha(t_{0})) \cdot \mathbf{u} - ((\alpha(t) - \alpha(t_{0})) \cdot \mathbf{u}) (\mathbf{u} \cdot \mathbf{u}) \\
										     & = (\alpha(t) - \alpha(t_{0})) \cdot \mathbf{u} - (\alpha(t) - \alpha(t_{0})) \cdot \mathbf{u} \\
										     & = 0.
				\end{split}
			\]
		Thus, $\sigma(t)$ does indeed lie in $\mathcal{P}$, the plane through $\alpha(t_{0})$ and orthogonal to $\mathbf{u}$.
		\item To show this, we compute $\sigma'$ and $\sigma''$. First, we compute $\sigma'$.
			\[
				\begin{split}
					\sigma'(t) & = \alpha'(t) - ((\alpha(t) - \alpha(t_{0}))' \cdot [\mathbf{u}, \alpha(t)] + (\alpha(t) - \alpha(t_{0})) \cdot [\mathbf{u}, \alpha(t)]') [\mathbf{u}, \alpha(t)] \\
						   & = \alpha'(t) - (\alpha'(t) \cdot [\mathbf{u}, \alpha(t)]) [\mathbf{u}, \alpha(t)] \\
				\end{split}
			\]
			Since $\alpha$ is a cylindrical helix, we know that $\mathbf{T} \cdot [\mathbf{u}, \alpha(t)]$ is a constant --- $\cos{\vartheta}$. Thus, $\alpha'(t) \cdot [\mathbf{u}, \alpha(t)] = v \cos{\vartheta}$. Also notice that $[\mathbf{u}, \alpha(t)] = \cos{\vartheta} \, \mathbf{T} + \sin{\vartheta} \, \mathbf{B}$ by our calculations in the previous problem. Thus,
			\[
				\begin{split}
					\sigma'(t) & = \alpha'(t) - v \cos{\vartheta} \, [\mathbf{u}, \alpha(t)] \\
						   & = v(1 - \cos^{2}{\vartheta}) \mathbf{T} - v \cos{\vartheta} \sin{\vartheta} \, \mathbf{B} \\
						   & = v \sin^{2}{\vartheta} \, \mathbf{T} - v \cos{\vartheta} \sin{\vartheta} \, \mathbf{B}
				\end{split}
			\]
			From this, we can calculate $\sigma''$.
			\[
				\begin{split}
					\sigma''(t) & = v \sin^{2}{\vartheta} \, \mathbf{T}' - v \cos{\vartheta} \sin{\vartheta} \, \mathbf{B}' + \frac{dv}{dt} \sigma'(t) \\
						    & = (\kappa v^{2} \sin^{2}{\vartheta} + \tau v^{2} \cos{\vartheta} \sin{\vartheta}) \, \mathbf{N} + \frac{dv}{dt} \sigma'(t) \\
						    & = (\kappa v^{2} \sin^{2}{\vartheta} + \kappa v^{2} \cos^{2}{\vartheta} ) \, \mathbf{N} + \frac{dv}{dt} \sigma'(t) \\
						    & = \kappa v^{2} \, \mathbf{N} + \frac{dv}{dt} \sigma'(t).
				\end{split}
			\]
			where we should note that $\tau \sin{\vartheta} = \kappa \cos{\vartheta}$ since $\alpha$ is a cylindrical helix.
			Thus, we can compute $\sigma' \times \sigma''$ as
			\[
				\begin{split}
					\sigma'(t) \times \sigma''(t) & = (v \sin^{2}{\vartheta} \, \mathbf{T} - v \cos{\vartheta} \sin{\vartheta} \, \mathbf{B}) \times \kappa v^{2} \, \mathbf{N} + \frac{dv}{dt} \sigma'(t) \times \sigma'(t) \\
								      & = (\kappa v^{3} \sin^{2}{\vartheta}) \, \mathbf{T} \times \mathbf{N} - (\kappa v^{3} \cos{\vartheta} \sin{\vartheta}) \, \mathbf{B} \times \mathbf{N} + 0 \\
								      & = \kappa v^{3} \cos{\vartheta} \sin{\vartheta} \, \mathbf{T} + \kappa v^{3} \sin^{2}{\vartheta} \, \mathbf{B}.
				\end{split}
			\]
			Thus,
			\[
				\begin{split}
					\lVert \sigma'(t) \times \sigma''(t) \rVert & = \kappa v^{3} \sin{\vartheta} \sqrt{\sin^{2}{\vartheta} + \cos^{2}{\vartheta}} \\
										    & = \kappa v^{3} \sin{\vartheta}
				\end{split}
			\]
			and
			\[
				\begin{split}
					\lVert \sigma'(t) \rVert & = v \sin{\vartheta}\sqrt{\sin^{2}{\vartheta} + \cos^{2}{\vartheta}} \\
								 & = v \sin{\vartheta}
				\end{split}
			\]
			Then, we get the curvature of the cross-sectional curve $\sigma$ as
			\[
				\begin{split}
					\text{curvature of } \sigma & = \frac{\kappa v^{3} \sin{\vartheta}}{(v \sin{\vartheta})^{3}} \\
								    & = \frac{\kappa}{\sin^{2}{\vartheta}}
				\end{split}
			\]
			as desired.
	\end{enumerate}
\end{solution}

\begin{problem}[Problem 4.]
	Consider the curve $\alpha : \mathbb{R} \to \mathbb{R}^{3}$  defined by $\alpha(t) = (2t, t^{2}, t^{3}/3)$ .
	\begin{enumerate}
		\item Verify that it is a cylindrical helix.
		\item Find the unit vector $\mathbf{u} = \mathbb{R}^{3}$.
		\item Find the angle $\vartheta$.
		\item Compute the cross-sectional curve $\sigma : \mathbb{R} \to \mathbb{R}^{3}$.
	\end{enumerate}
\end{problem}

\begin{solution}
	\begin{enumerate}
		\item Note that since we have $\mathbf{T} = \left [ \left ( \frac{2}{t^{2} + 2}, \frac{2t}{t^{2} + 2}, 1 - \frac{2}{t^{2} + 2} \right ), \alpha(t) \right ]$, we have that the dot product with any tangent vector of the form $[(a, 0, a), \alpha(t)]$ (for some $a \in \mathbb{R}$) is constant ---
			\[
				\begin{split}
					\mathbf{T} \cdot [(a, 0, a), \alpha(t)] & = \frac{2a}{t^{2} + 2} + a - \frac{2a}{t^{2} + 2} \\
										& = a.
				\end{split}
			\]
		We can choose $a$ so that $[(a, 0, a), \alpha(t)]$ such that $\lVert [(a, 0, a), \alpha(t)] \rVert = 1$. Specifically, for $2a^{2} = 1$ and thus $a = \pm \frac{1}{\sqrt 2}$ we have the desired norm. Thus, we have a constant unit tangent vector $[\mathbf{u}, \alpha(t)]$ such that $\mathbf{T} \cdot [\mathbf{u}, \alpha(t)]$ is constant.
		\item 
			\[
				\boxed{
				\mathbf{u} = \left ( \frac{1}{\sqrt 2}, 0, \frac{1}{\sqrt 2} \right )
				}
			\]
		\item 
			\[
				\boxed{
					\vartheta = \frac{\pi}{4}
				}
			\] 
			We have by our computation in part (a) that $\mathbf{T} \cdot [\mathbf{u}, \alpha(t)] = \frac{1}{\sqrt 2}$. By the definition of $\vartheta$, we have then that $\cos{\vartheta} = \frac{1}{\sqrt 2}$. Thus,
			\[
				\begin{split}
					\vartheta & = \arccos{\frac{1}{\sqrt 2}} \\
						  & = \frac{\pi}{4}.
				\end{split}
			\]
		\item
			\[
				\boxed{
					\sigma(t) = (2t, t^{2}, t^{3}/3) - \left ( t - t_{0} + \frac{t^{3} - t_{0}^{3}}{6}, 0, t - t_{0} + \frac{t^{3} - t_{0}^{3}}{6} \right ).
				}
			\]
			For some given $t_{0} \in \mathbb{R}$ we have the cross sectional curve given by
			 \[
				\begin{split}
					\sigma(t) & = \alpha(t) - ((\alpha(t) - \alpha(t_{0})) \cdot \mathbf{u}) \mathbf{u} \\
						  & = (2t, t^{2}, t^{3}/3) - ((2t - 2t_{0}, t^{2} - t_{0}^{2}, t^{3}/3 - t_{0}^{3}/3) \cdot (1/\sqrt 2, 0, 1/\sqrt 2))(1/\sqrt 2, 0, 1/\sqrt 2) \\
						  & = (2t, t^{2}, t^{3}/3) - \left ( \sqrt 2(t - t_{0}) + \frac{t^{3} - t_{0}^{3}}{3 \sqrt 2} \right ) (1/\sqrt 2, 0, 1/\sqrt 2) \\
						  & = (2t, t^{2}, t^{3}/3) - \left ( t - t_{0} + \frac{t^{3} - t_{0}^{3}}{6}, 0, t - t_{0} + \frac{t^{3} - t_{0}^{3}}{6} \right ).
				\end{split}
			\]
	\end{enumerate}
\end{solution}

\begin{problem}[Problem 5.]
	Prove that the cubic curve $\alpha(t) = (at, b t^{2}, c t^{3})$ for some $a, b, c \in \mathbb{R} \setminus \{ 0 \}$ is a cylindrical helix if and only if $3ac = \pm 2b^{2}$.
\end{problem}

\begin{solution}
	By definition $\alpha$  is a cylindrical helix if and only if we have a constant unit tangent vector $[\mathbf{u}, \alpha(t)]$  such that $\mathbf{T} \cdot [\mathbf{u}, \alpha(t)] = \cos{\vartheta}$ for some constant angle $\vartheta$ . The following chain of equivalences follows ---
	\[
		\begin{split}
			& \mathbf{T} \cdot [\mathbf{u}, \alpha(t)] = \cos{\vartheta} \\
			\iff & \frac{\alpha'(t)}{\lVert \alpha'(t) \rVert} \cdot \mathbf{u} = \cos{\vartheta} \\
			\iff & \alpha'(t) \cdot \mathbf{u} = \lVert \alpha'(t) \rVert \cos{\vartheta} \\
			\iff & (\alpha'(t) \cdot \mathbf{u})^{2} = \lVert \alpha'(t) \rVert^{2} \cos^{2}{\vartheta} \\
			\iff & (\alpha'(t) \cdot \mathbf{u})^{2} - \lVert \alpha'(t) \rVert^{2} \cos^{2}{\vartheta} = 0.
		\end{split}
	\]
	for all $t \in I$. Let us have $\gamma$ defined by $\gamma(t) = (\alpha'(t) \cdot \mathbf{u})^{2} - \lVert \alpha'(t) \rVert^{2} \cos^{2}{\vartheta}$. Then what we want to prove is equivalent to showing that $\gamma$ is identically zero on $I$. \\\\
	Note that
	\[
		\alpha'(t) = [(a, 2b t, 3 c t^{2}), \alpha(t)]
	\]
	and thus,
	\[
		\lVert \alpha'(t) \rVert^{2} = a^{2} + 4b^{2} t^{2} + 9c^{2} t^{4}. 
	\]
	Thus, for $\mathbf{u} = (u_{1}, u_{2}, u_{3})$ we have
	\[
		\begin{split}
			\gamma(t) & = (a u_{1} + 2 b u_{2} t + 3 c u_{3} t^{2})^{2} - \cos^{2}{\vartheta}(a^{2} + 4b^{2} t^{2} + 9 c^{2} t^{4}) \\
				  & = a^{2}(u_{1}^{2} - \cos^{2}{\vartheta}) - 4b^{2}t^{2}(u_{2}^{2} - \cos^{2}{\vartheta}) + 9 c^{2} t^{4} (u_{3}^{2} - \cos^{2}{\vartheta}) \\ 
				  & + 2(2 ab u_{1} u_{2} t + 6 bc u_{2} u_{3} t^{3} + 3 ca u_{3} u_{1} t^{2}) \\
				  & = a^{2}(u_{1}^{2} - \cos^{2}{\vartheta}) - t^{2}(4 b^{2} (u_{2}^{2} - \cos^{2}{\vartheta}) + 6 ac u_{3} u_{1}) + 9 c^{2} t^{4} (u_{3}^{2} - \cos^{2}{\vartheta}) \\ 
				  & + 4 ab u_{1} u_{2} t + 12 bc u_{2} u_{3} t^{3}) \\
		\end{split}
	\]
	Since $\gamma$ is a polynomial in $t$, $\gamma$ being identically zero is equivalent to all of its coefficients being zero. Note that for the $t^{0}$ coefficient, $a^{2}(u_{1}^{2} - \cos^{2}{\vartheta})$ to be zero, we need $u_{1} = \pm \cos{\vartheta}$ since $a \neq 0$. Similarly, for the $t^{4}$ coefficient, $9c^{2}(u_{3}^{2} - \cos^{2}{\vartheta})$ to be zero, we need $u_{3} = \pm \cos{\vartheta}$. Since we can have $\vartheta \neq \pi/2$ (for a specific example, in the previous problem we have $\vartheta = \pi/4$) and thus, $u_{1} \neq 0$ or $u_{3} \neq 0$ we need $u_{2} = 0$ to ensure that the $t$ and $t^{3}$ coefficients are zero. Notice that all of these are also sufficient to ensure that the coefficients are zero, and thus, are equivalent to the coefficients being zero. \\\\
	Thus, the $t^{2}$ coefficient being $0$ is equivalent to 
	\[
		6ac u_{3} u_{1} - 4 b ^{2} \cos^{2}{\vartheta} = 0
	\]
	Since we can have $\cos{\vartheta} \neq 0$, this is equivalent to
	\[
		6ac = \pm 4b^{2}
	\]
	or
	\[
		3ac = \pm 2b^{2}
	\]
	where the sign depends on whether $u_{1}$ and $u_{3}$ have the same sign. This is the desired result.
\end{solution}
