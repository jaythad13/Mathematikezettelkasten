\begin{problem}[Problem 1.]
	Let $\mathbf V_1 = \mathbf U_1 - x \mathbf U_3$, $\mathbf V_2 = \mathbf U_2$, and $\mathbf V_3 = x \mathbf U_1 + \mathbf U_3$ in terms of the \textit{natural frame field} on $\mathbb R^3$.
	\begin{enumerate}
		\item Prove that the vectors $\mathbf V_1 (\mathbf p), \mathbf V_2(\mathbf p), \mathbf V_3(\mathbf p)$ are linearly independent at each point $\mathbf p \in \mathbb R^3$.
		\item Express the vector field $\mathbf V  = x \mathbf U_1 + y \mathbf U_2 + z \mathbf U_3$ as a linear combination of $\mathbf V_1$, $\mathbf V_2$, and $\mathbf V_3$.
	\end{enumerate}
\end{problem}

\begin{solution}
	\begin{enumerate}
		\item 
			We can calculate the vectors explicitly as follows.
			\[
				\begin{gathered}
					\begin{split}
						\mathbf V_1(\mathbf p) & = \mathbf U_1(\mathbf p) - x \mathbf U_3(\mathbf p) \\
								       & = [(1, 0, 0), \mathbf p] - [x (0, 0, 1), \mathbf p] \\
								       & = [(1, 0, - x), \mathbf p].
					\end{split}\\\\
					\begin{split}
						\mathbf V_2(\mathbf p) & = \mathbf U_2(\mathbf p) \\
								       & = [(0, 1, 0), \mathbf p].
					\end{split}\\\\
					\begin{split}
						\mathbf V_3(\mathbf p) & = x \mathbf U_1(\mathbf p) + \mathbf U_3(\mathbf p) \\
								       & = [x (1, 0, 0), \mathbf p] + [(0, 0, 1), \mathbf p] \\
								       & = [(x, 0, 1), \mathbf p].
					\end{split}
				\end{gathered}
			\]\\
			Suppose scalars $a_1, a_2, a_3 \in \mathbb R$ give us $a_1 \mathbf V_1(\mathbf p) + a_2 \mathbf V_2(\mathbf p) + a_3 \mathbf V_3(\mathbf p) = [\mathbf 0, \mathbf p]$ (note that the right hand side is the additive identity in $\mathrm T_{\mathbf p} \mathbb R^3$). Then we must have
			\[
				[(a_1 + a_3 \, x, a_2, - a_1 \, x + a_3)] = [(0, 0, 0), \mathbf p]
			\]
			giving us the system of equations
			\[
				\begin{gathered}
					a_1 + a_3 \, x = 0 \\
					a_2 = 0 \\
					-a_1 \, x + a_3 = 0
				\end{gathered}
			\]
			which we can write as a matrix equation
			\[
				\begin{bmatrix}
					1 & 0 & x \\
					0 & 1 & 0 \\
					-x & 0 & 1
					\end{bmatrix} \begin{bmatrix}
					a_1 \\
					a_2 \\
					a_3
				\end{bmatrix} = \mathbf 0
			\]
			Since the matrix has determinant $x^2 + 1$, its determinant is non-zero for all $x \in \mathbb R$. Then the matrix is always invertible, the only solution to the equation is $a_1 = a_2 = a_3 = 0$, regardless of $x$. Thus, $\mathbf V_1(\mathbf p), \mathbf V_2(\mathbf p), \mathbf V_3(\mathbf p)$ is a linearly independent list of vectors in $\mathrm T_{\mathbf p} \mathbb R^3$ for all $\mathbf p \in \mathbb R^3$.
		\item
			\[
				\boxed{
					\mathbf V = \frac{x - zx}{x^2 + 1} \mathbf V_1 + y \mathbf V_2 + \frac{x^2 + z}{x^2 + 1} \mathbf V_3.
				}
			\] \\\\
			Note that we're thinking of the vector fields as elements of a module over the ring $\mathcal C^\infty(\mathbb R^3)$ with the obvious definitions of addition and scalar multiplication. \\\\
			We notice that $-x \mathbf V_1 + \mathbf V_3 = (x^2 + 1) \mathbf U_3$, and thus, we can get
			\[
				\mathbf U_3 = \frac{-x}{x^2 + 1} \mathbf V_1 + \frac{1}{x^2 + 1} \mathbf V_3
			\]
			as a well defined linear combination for all inputs $\mathbf p \in \mathbb R^3$. Similarly, $\mathbf V_1 + x \mathbf V_3 = (x^2 + 1) \mathbf U_1$ allows us to write the linear combination
			\[
				\mathbf U_1 = \frac{1}{x^2 + 1} \mathbf V_1 + \frac{x}{x^2 + 1} \mathbf V_3
			\]
			Now that we have $\mathbf U_1$ and $\mathbf U_3$ (and $\mathbf U_2 = \mathbf V_2$) as linear combinations of $\mathbf V_1, \mathbf V_2, \mathbf V_3$, we can just write
			\[
				\begin{split}
					\mathbf V & = x \mathbf U_1 + y \mathbf U_2 + z \mathbf U_3 \\
						  & = x \left ( \frac{1}{x^2 + 1} \mathbf V_1 + \frac{x}{x^2 + 1} \mathbf V_3 \right ) + y \mathbf V_2 + z \left ( \frac{-x}{x^2 + 1} \mathbf V_1 + \frac{1}{x^2 + 1} \mathbf V_3 \right ) \\
						  & = \frac{x - zx}{x^2 + 1} \mathbf V_1 + y \mathbf V_2 + \frac{x^2 + z}{x^2 + 1} \mathbf V_1
				\end{split}
			\]
	\end{enumerate}
\end{solution}

\begin{problem}[Problem 2.]
	Let $\mathbf V = y^2 \mathbf U_1 - x \mathbf U_3$, and let $f(x, y, z) = xy$ and $g(x, y, z) = z^3$. Compute the following functions.
	\begin{enumerate}
		\item $\mathbf V[f]$
		\item $\mathbf V[g]$
		\item $\mathbf V[fg]$
		\item $f \mathbf V[g] - g \mathbf V[f]$
		\item $\mathbf V[f^2 + g^2]$
		\item $\mathbf V[\mathbf V[f]]$
	\end{enumerate}
\end{problem}

\begin{solution}
	Let $\mathbf p$ be an arbitrary point $(x, y, z) \in \mathbb R^3$. First, we can evaluate $\mathbf V$ generally ---
	\[
		\mathbf V(\mathbf p) = [(y^2, 0, -x), \mathbf p].
	\]
	Let $\mathbf v(\mathbf p) = (y^2, 0, -x)$. We also evaluate the gradients of $f$ and $g$ ---
	\[
		\nabla f(\mathbf p) = (y, x, 0)	
	\]
	and
	\[
		\nabla g(\mathbf p) = (0, 0, 3z^2)
	\]
	Now we can compute the functions. We recall that $\mathbf V [f]$ is a function given by $\mathbf p \mapsto \nabla f(\mathbf p) \cdot \mathbf V(\mathbf p)$.
	\begin{enumerate}
		\item 
			\[
				\boxed{
					\mathbf V[f](\mathbf p) = y^3.
				}
			\] \\
			We can just compute explicitly ---
			\[
				\begin{split}
					\mathbf V[f](\mathbf p) & = \nabla f(\mathbf p) \cdot \mathbf v(\mathbf p) \\
								& = (y, x, 0) \cdot (y^2, 0, -x) \\
								& = y^3.
				\end{split}
			\]
		\item
			\[
				\boxed{
					\mathbf V[g](\mathbf p) = - 3 x z^2
				}
			\] \\
			Again, we just compute explicitly
			\[
				\begin{split}
					\mathbf V[g](\mathbf p) & = \nabla g(\mathbf p) \cdot \mathbf v(\mathbf p) \\
								& = (0, 0, 3z^2) \cdot (y^2, 0, -x) \\
								& = - 3x z^2.
				\end{split}
			\]
		\item
			\[
				\boxed{
					\mathbf V[fg](\mathbf p) = y^3 z^3 - 3 x^2 y z^2
				}
			\] \\
			Note that here, we use the product rule for gradients --- $\nabla(fg) = (\nabla f) g + f (\nabla g)$
			\[
				\begin{split}
					\mathbf V[fg](\mathbf p) & = \nabla(fg) \cdot \mathbf v(\mathbf p) \\
								 & = ((\nabla f)(\mathbf p) g(\mathbf p) + f(\mathbf p) (\nabla g)(\mathbf p)) \cdot \mathbf v(\mathbf p) \\
								 & = g(\mathbf p) (\nabla f)(\mathbf p) \cdot \mathbf v(\mathbf p) + f(\mathbf p) (\nabla g(\mathbf p)) \cdot \mathbf v(\mathbf p) \\
								 & = z^3 y^3 - (xy)3xz^2 \\
								 & = y^3 z^3 - 3 x^2 y z^2.
				\end{split}
			\]
		\item
			\[
				\boxed{
					(f \mathbf V[g] + g \mathbf V[f])(\mathbf p) = y^3 z^3 - 3 x^2 y z^2
				}
			\] \\
			Here we can use our previous computations of $\mathbf V[f](\mathbf p)$ and $\mathbf V[g](\mathbf p)$ to get
			\[
				\begin{split}
					(f \mathbf V[g] + g \mathbf V[f])(\mathbf p) & = f(\mathbf p) \mathbf V[g](\mathbf p) + g(\mathbf p) \mathbf V[f](\mathbf p) \\
										     & = (xy)(-3xz^2) + (z^3)(y^3) \\
										     & = y^3 z^3 - 3 x^2 y z^2
				\end{split}
			\]
		\item
			\[
				\boxed{
					\mathbf V[f^2 + g^2](\mathbf p) = 2 x y^4 - 6 x z^5.
				}
			\] \\
			First, we compute $\nabla (f^2 + g^2)$. Here, again we can use the sum and product rules for the gradient.
			\[
				\begin{split}
					\nabla(f^2 + g^2)(\mathbf p) & = \nabla(f^2)(\mathbf p) + \nabla(g^2)(\mathbf p) \\
								     & = 2 f(\mathbf p) \nabla f(\mathbf p) + 2 g(\mathbf p) \nabla f(\mathbf p) \\
								     & = 2 ((xy)(y, x, 0) + (z^3)(0, 0, 3z^2)) \\
								     & = 2 (x y^2, x^2 y, 3z^5).
				\end{split}
			\] \\
			Then we get
			\[
				\begin{split}
					\mathbf V[f^2 + g^2](\mathbf p) & = \nabla(f^2 + g^2)(\mathbf p) \cdot \mathbf v(\mathbf p) \\
									& = 2 (x y^2, x^2 y, 3z^5) \cdot (y^2, 0, -x) \\
									& = 2 x y^4 - 6 x z^5.
				\end{split}
			\]
		\item
			\[
				\boxed{
					\mathbf V[\mathbf V[f]] (\mathbf p) = 0.
				}
			\] \\
			Let $h : \mathbb R^3 \to \mathbb R$ be given by $\mathbf p \mapsto \mathbf V[f](\mathbf p) = y^3$. Note that $\nabla h(\mathbf p) = (0, 3y^2, 0)$. Then we can calculate
			\[
				\begin{split}
					\mathbf V[\mathbf V[f]](\mathbf p) & = \mathbf V[h](\mathbf p) \\
									   & = \nabla h(\mathbf p) \cdot \mathbf v(\mathbf p) \\
									   & = (0, 3y^2, 0) \cdot (y^2, 0, -x) \\
									   & = 0.
				\end{split}
			\]
	\end{enumerate}
\end{solution}

\begin{problem}[Problem 3.]
	Let $\mathbf V : \mathbb R^3 \to \mathrm T \mathbb R^3$ denote a vector field. Show that we have the identity
	\[
		\mathbf V = \mathbf V[x_1] \mathbf U_1 + \mathbf V[x_2] \mathbf U_2 + \mathbf V[x_3] \mathbf U_3
	\]
\end{problem}

\begin{solution}
	For $\mathbf p = (x, y, z) \in \mathbb R^3$, we have $x_1(\mathbf p) = x$, $x_2(\mathbf p) = y$, and $x_3(\mathbf p) = z$. Note that $\nabla x_i = \mathbf e_i$, where $\mathbf e_1, \mathbf e_2, \mathbf e_3$ is the standard orthonormal basis on $\mathbb R^3$. \\\\
	Let $\mathbf V(\mathbf p) = [\mathbf v(\mathbf p), \mathbf p]$. Then, since $\mathbf v$ is a function $\mathbb R^3 \to \mathbb R^3$, we can decompose it into component functions $v_1, v_2, v_3 : \mathbb R^3 \to \mathbb R$ such that $\mathbf v(\mathbf p) = (v_1(\mathbf p), v_2(\mathbf p), v_3(\mathbf p))$. Thus, we can evaluate $\mathbf V[x_i]$ as
	\[
		\begin{split}
			\mathbf V[x_i](\mathbf p) & = \nabla x_i(\mathbf p) \cdot \mathbf V(\mathbf p) \\
						  & = \mathbf e_i \cdot (v_1(\mathbf p), v_2(\mathbf p), v_3(\mathbf p)) \\
						  & = v_i(\mathbf p)
		\end{split}
	\]
	at any point $\mathbf p \in \mathbb R^3$. \\\\
	Thus, we get
	\[
		\begin{split}
			& \mathbf V[x_1] (\mathbf p) \mathbf U_1 (\mathbf p) + \mathbf V[x_2] (\mathbf p) \mathbf U_2 (\mathbf p) + \mathbf V[x_3] (\mathbf p) \mathbf U_3 (\mathbf p) \\
			& = v_1 (\mathbf p) \mathbf U_1 (\mathbf p) + v_2 (\mathbf p) \mathbf U_2 (\mathbf p) + v_3 (\mathbf p) \mathbf U_3 (\mathbf p) \\
			& = [v_1(\mathbf p)(1, 0, 0), \mathbf p] + [v_2 (\mathbf p) (0, 1, 0), \mathbf p] + [v_3(\mathbf p)(0, 0, 1), \mathbf p] \\
			& = [(v_1(\mathbf p), v_2(\mathbf p), v_3(\mathbf p)), \mathbf p] \\
			& = \mathbf V(\mathbf p)
		\end{split}
	\]
	for all $\mathbf p \in \mathbb R^3$. That is, we have
	\[
		\mathbf V = \mathbf V[x_1] \mathbf U_1 + \mathbf V[x_2] \mathbf U_2 + \mathbf V[x_3] \mathbf U_3.
	\]
\end{solution}

\begin{problem}[Problem 4.]
	Let $\mathbf V : \mathbb R^3 \to \mathrm T \mathbb R^3$ denote a vector field such that $\mathbf v = (v_1, v_2, v_3)$ corresponding to $\mathbf V(\mathbf p) = [\mathbf v, \mathbf p]$ is independent of $\mathbf p$. Show that, for any smooth function $f : \mathbb R^3 \to \mathbb R$, we have the following identities:
	\[
		\mathbf V[f] = \sum_{i = 1}^3 v_i \frac{\partial f}{\partial x_i}
	\]
	and
	\[
		\mathbf V[\mathbf V[f]] = \sum_{i = 1}^3 \sum_{j = 1}^3 v_i v_j \frac{\partial^2 f}{\partial x_i \partial x_j}.
	\]
\end{problem}

\begin{solution}
	We know that 
	\[
		\nabla f (\mathbf p) = \left ( \frac{\partial f}{\partial x_1}(\mathbf p), \frac{\partial f}{\partial x_2}(\mathbf p), \frac{\partial f}{\partial x_3}(\mathbf p) \right ).
	\] \\
	Thus,
	\[
		\begin{split}
			\mathbf V[f](\mathbf p) & = \nabla f(\mathbf p) \cdot \mathbf v \\
						& = \sum_{j = 1}^3 v_j \frac{\partial f}{\partial x_j}(\mathbf p).
		\end{split}
	\] \\
	Thus, we can write
	\[
		\begin{split}
			& \nabla (\mathbf V[f])(\mathbf p) \\
			& = \left ( \frac{\partial (\mathbf V[f])}{\partial x_1} \Big |_{\mathbf p}, \frac{\partial (\mathbf V[f])}{\partial x_2} \Big |_{\mathbf p}, \frac{\partial (\mathbf V[f])}{\partial x_3} \Big |_{\mathbf p} \right ) \\
			& = \left ( \frac{\partial}{\partial x_1} \left ( \sum_{j = 1}^3 v_j \frac{\partial f}{\partial x_j} \Big |_{\mathbf p} \right ) \Big |_{\mathbf p}, \frac{\partial}{\partial x_2} \left ( \sum_{j = 1}^3 v_j \frac{\partial f}{\partial x_j} \Big |_{\mathbf p} \right ) \Big |_{\mathbf p}, \frac{\partial}{\partial x_3} \left ( \sum_{j = 1}^3 v_j \frac{\partial f}{\partial x_j} \Big |_{\mathbf p} \right ) \Big |_{\mathbf p} \right ).
		\end{split}
	\] \\
	Finally, we have
	\[
		\begin{split}
			\mathbf V[\mathbf V[f]](\mathbf p) & = \nabla(\mathbf V[f])(\mathbf p) \cdot \mathbf v \\
			& = \left ( \frac{\partial}{\partial x_1} \left ( \sum_{j = 1}^3 v_j \frac{\partial f}{\partial x_j} \right ) \Big |_{\mathbf p}, \frac{\partial}{\partial x_2} \left ( \sum_{j = 1}^3 v_j \frac{\partial f}{\partial x_j} \right ) \Big |_{\mathbf p}, \frac{\partial}{\partial x_3} \left ( \sum_{j = 1}^3 v_j \frac{\partial f}{\partial x_j} \right ) \Big |_{\mathbf p} \right ) \cdot \mathbf v \\
			& = \sum_{i = 1}^3 v_i \frac{\partial f}{\partial x_i} \left ( \sum_{j = 1}^3 v_j \frac{\partial f}{\partial x_j} \right ) \Big |_{\mathbf p} \\
			& = \sum_{i = 1}^3 \sum_{j = 1}^3 v_i v_j \frac{\partial^2 f}{\partial x_i \partial x_j} \Big |_{\mathbf p}.
		\end{split}
	\]
	for every $\mathbf p \in \mathbb R^3$. \\\\
	That is, we have 
	\[
		\mathbf V[\mathbf V[f]] = \sum_{i = 1}^3 \sum_{j = 1}^3 v_i v_j \frac{\partial^2 f}{\partial x_i \partial x_j}
	\]
	as desired.
\end{solution}

\begin{problem}[Problem 5.]
	Say $\mathbf V, \mathbf W : \mathbb R^3 \to \mathrm T \mathbb R^3$ are two vector fields. Show that $\mathbf V = \mathbf W$ if and only if $\mathbf V[f] = \mathbf W[f]$ for every differentiable function $f : \mathbb R^3 \to \mathbb R$.
\end{problem}

\begin{solution}
	Let $\mathbf v(\mathbf p)$ and $\mathbf w(\mathbf p)$ denote the vector parts of $\mathbf V(\mathbf p)$ and $\mathbf W(\mathbf p)$ respectively. \\\\
	Suppose that $\mathbf V = \mathbf W$. Then, obviously, for at each point $\mathbf p \in \mathbb R^3$, we have $\mathbf v(\mathbf p) = \mathbf w(\mathbf p)$, and thus, 
	\[
		\begin{split}
			\mathbf V[f](\mathbf p) & = \nabla f(\mathbf p) \cdot \mathbf v(\mathbf p) \\
						& = \nabla f(\mathbf p) \cdot \mathbf w(\mathbf p) \\
						& = \mathbf W[f](\mathbf p).
		\end{split}
	\] \\
	Since we have $\mathbf V[f](\mathbf p) = \mathbf W[f](\mathbf p)$ for all $\mathbf p \in \mathbb R^3$, we can say $\mathbf V[f] = \mathbf W[f]$. \\\\
	To prove the converse, suppose that we have $\mathbf V[f] = \mathbf W[f]$ for all differentiable functions $f : \mathbb R^3 \to \mathbb R$. Then pointwise, we have
	\[
		\mathbf V[f](\mathbf p) = \mathbf W[f](\mathbf p)
	\]
	or,
	\[
		\nabla f(\mathbf p) \cdot \mathbf v (\mathbf p) = \nabla f(\mathbf p) \cdot \mathbf w (\mathbf p)
	\]
	at every point $\mathbf p \in \mathbb R^3$. We rewrite this as
	\[
		\nabla f(\mathbf p) \cdot (\mathbf v(\mathbf p) - \mathbf w(\mathbf p)) = 0.
	\] \\
	Note that we can construct $f : \mathbf p \to a x + b y + c z$ with constant gradient $(a, b, c)$ for any desired $(a, b, c) \in \mathbb R^3$. That means, that at each point $\mathbf p \in \mathbb R^3$, we must have
	\[
		(a, b, c) \cdot (\mathbf v(\mathbf p) - \mathbf w(\mathbf p)) = 0
	\]
	for every $(a, b, c) \in \mathbb R^3$, including $(a, b, c) = (\mathbf v(\mathbf p) - \mathbf w(\mathbf p))$. Thus, at each point $\mathbf p \in \mathbb R^3$ we must have
	\[
		\lVert \mathbf v(\mathbf p) - \mathbf w(\mathbf p) \rVert^2 = 0
	\]
	and thus, $\mathbf v(\mathbf p) = \mathbf w(\mathbf p)$.\\\\
	Finally, this gives us
	\[
		\mathbf V(\mathbf p) = [\mathbf v(\mathbf p), \mathbf p] = [\mathbf w(\mathbf p), \mathbf p] = \mathbf W(\mathbf p)
	\]
	for all $\mathbf p \in \mathbf R^3$, or just $\mathbf V = \mathbf W$, as desired.
\end{solution}
