\begin{problem}[4.]
	Let $g$ be a continuous real-valued function on the unit circle $S^{1} = \{ \mathbf{x} \in \mathbb{R}^{2}\mid \lVert \mathbf{x} \rVert = 1 \}$ such that $g(0, 1) = g(1, 0) = 0$ and $g(-\mathbf{x}) = - g(\mathbf{x})$. Define $f : \mathbb{R}^{2} \to \mathbb{R}$ by
	\[
		f(\mathbf{x}) = \begin{cases}
			\lVert \mathbf{x} \rVert \, g \left ( \dfrac{\mathbf{x}}{\lVert \mathbf{x} \rVert} \right ) & \mathbf{x} \neq \mathbf{0} \\
			0 & \mathbf{x} = \mathbf{0}.
		\end{cases}
	\]
	\begin{enumerate}
		\item If $\mathbf{x} \in \mathbb{R}^{2}$ and $h : \mathbb{R} \to \mathbb{R}$ is defined by $h(t) = f(t\mathbf{x})$, show that $h$ is differentiable.
		\item Show that $f$ is not differentiable at $(0, 0)$ unless $g = 0$.
	\end{enumerate}
\end{problem}

\begin{problem}[12.]
	A function $f : \mathbb{R}^{n} \times \mathbb{R}^{m} \to \mathbb{R}^{p}$ is \textit{bilinear} if for $\mathbf{x}_{1}, \mathbf{x}_{2} \in \mathbb{R}^{n}$, $\mathbf{y}_{1}, \mathbf{y}_{2} \in \mathbb{R}^{m}$ and $a \in \mathbb{R}$ we have
	\[
		\begin{split}
			f(a \mathbf{x}_{1}, \mathbf{y}_{1}) & = a f(\mathbf{x}_{1}, \mathbf{y}_{1}) = f(\mathbf{x}_{1}, a \mathbf{y}_{1}) \\
			f(\mathbf{x}_{1} + \mathbf{x}_{2}, \mathbf{y}_{1}) & = f(\mathbf{x}_{1}, \mathbf{y}) + f(\mathbf{x}_{2}, \mathbf{y}_{1}) \\
			f(\mathbf{x}_{1}, \mathbf{y}_{1} + \mathbf{y}_{2}) & = f(\mathbf{x}_{1}, \mathbf{y}_{1}) + f(\mathbf{x}_{1}, \mathbf{y}_{2}).
		\end{split}
	\]
	\begin{enumerate}
		\item Prove that if $f$ is bilinear, then
			\[
				\lim_{(\mathbf{h}, \mathbf{k}) \to \mathbf{0}} \frac{\lVert f(\mathbf{h}, \mathbf{k}) \rVert}{\lVert (\mathbf{h}, \mathbf{k}) \rVert} = 0.
			\]
		\item Prove that $Df \Big |_{(\mathbf{a}, \mathbf{b})} (\mathbf{x}, \mathbf{y}) = f(\mathbf{a}, \mathbf{y}) + f(\mathbf{x}, \mathbf{b})$.
		\item Show that the formula for $D \Pi$ where $\Pi : \mathbb{R}^{2} \to \mathbb{R}$ is the product function $(x, y) \mapsto xy$ is a special case of (b).
	\end{enumerate}
\end{problem}

\begin{problem}[21.]
	Let $g_{1}, g_{2} : \mathbb{R}^{2} \to \mathbb{R}$ be continuous. Define $f : \mathbb{R}^{2} \to \mathbb{R}$ by
	\[
		f(x, y) = \int_{0}^{x} g_{1}(t, 0) \, dt + \int_{0}^{y} g_{2}(x, t) \, dt.
	\]
	\begin{enumerate}
		\item Show that $D_{2} f \Big |_{(x, y)} = g_{2}(x, y)$.
		\item How should $f$ be defined so that $D_{1} f \Big |_{(x, y)} = g_{1}(x, y)$?
		\item Find a function $f : \mathbb{R}^{2} \to \mathbb{R}$ such that $D_{1} f(x, y) = x$ and $D_{2} f(x, y) = y$. Find one such that $D_{1} f(x, y) = y$ and $D_{2} f(x, y) = x$.
	\end{enumerate}
\end{problem}

\begin{problem}[26.]
	Let
	 \[
		f(x) = \begin{cases}
			e^{-(x - 1)^{2}} e^{-(x + 1)^{2}} & x \in (-1, 1) \\
			0 & x = 0.
		\end{cases}
	\]
	\begin{enumerate}
		\item Show that $f : \mathbb{R} \to \mathbb{R}$ is a $\mathcal{C}^{\infty}$ functions which is positive on $(-1, 1)$ and $0$ elsewhere.
		\item Show that there is a $\mathcal{C}^{\infty}$ function $g : \mathbb{R} \to [0, 1]$ such that $g(x) = 0$ for $x \le 0$ and $g(x) = 1$ for $x \ge \varepsilon$.
		\item If $\mathbf{a} \in \mathbb{R}^{n}$ define $g : \mathbb{R}^{n} \to \mathbb{R}$ by
			\[
				g(\mathbf{x}) = f((x_{1} - a_{1})/\varepsilon) \cdots f((x_{n} - a_{n})/\varepsilon).
			\]
			Show that $g$ is a $\mathcal{C}^{\infty}$ function which is positive on $(a_{1} - \varepsilon, a_{1} + \varepsilon) \times \dots \times (a_{n} - \varepsilon, a_{n} + \varepsilon)$ and zero elsewhere.
		\item If $A \subset \mathbb{R}^{n}$ is open and $C \subset A$ is compact, show that there is a non-negative $\mathcal{C}^{\infty}$ function $f : A \to \mathbb{R}$ such that $f(\mathbf{x}) > 0$ for $\mathbf{x} \in C$ and $f = 0$ outside of some closed set contained in $A$.
		\item Show that we can choose such an $f$ so that $f: A \to [0, 1]$ and $f(\mathbf{x}) = 1$ for $\mathbf{x} \in C$.
	\end{enumerate}
\end{problem}

\begin{problem}[29.]
	Let $f : \mathbb{R}^{n} \to \mathbb{R}$. For $\mathbf{x} \in \mathbb{R}^{n}$, the limit
	\[
		\lim_{t \to 0} \frac{f(\mathbf{a} + t \mathbf{v}) - f(\mathbf{a})}{t},
	\]
	if it exists, is denoted $D_{\mathbf{v}} f \Big |_{\mathbf{a}}$, and called the \textit{directional derivative} of $f$ at $\mathbf{a}$, in the direction $\mathbf{v}$.
	\begin{enumerate}
		\item Show that $D_{\mathbf{e}_{i}} f \Big |_{\mathbf{a}} = D_{i} f \Big |_{\mathbf{a}}$.
		\item Show that $D_{t\mathbf{v}} f \Big |_{\mathbf{a}} = t D_{\mathbf{v}} f \Big |_{\mathbf{a}}$.
		\item If $f$ is differentiable at $\mathbf{a}$, show that $D_{\mathbf{v}} f \Big |_{\mathbf{a}} = Df \Big |_{\mathbf{a}} (\mathbf{v})$ and therefore $D_{\mathbf{v} + \mathbf{w}} f \Big |_{\mathbf{a}} = D_{\mathbf{v}} f \Big |_{\mathbf{a}} + D_{\mathbf{w}} f \Big |_{\mathbf{a}}$.
	\end{enumerate}
\end{problem}
